33 /// Describe a conversion from `Angle` to a 3D vector.
34 /// Sweep from the first axis towards the second axis.
35 /// For example, for kAngleToVectorXY,
36 /// angle = 0 degrees ==> x-axis (1, 0, 0)
37 /// angle = 90 degrees ==> y-axis (0, 1, 0)
38 /// angle = 180 degrees ==> negative x-axis (-1, 0, 0)
39 /// angle = 90 degrees ==> negative y-axis (0, -1, 0)
40 /// Note that we don't have to worry about handedness. No matter the handedness
41 /// of the coordinate system, we always sweep from one axis towards the other.
42 /// That is, the other axis is always at 90 degrees, never -90.
43 enum AngleToVectorSystem {
44  kAngleToVectorXY,
45  kAngleToVectorXZ,
46  kAngleToVectorYZ,
47  kAngleToVectorYX,
48  kAngleToVectorZX,
49  kAngleToVectorZY,
50  kAngleToVectorCount
51 };

53 static const float kPi = static_cast<float>(M_PI);
54 static const float kTwoPi = static_cast<float>(2.0 * M_PI);
55 static const float kThreePi = static_cast<float>(3.0 * M_PI);
56 static const float kHalfPi = static_cast<float>(M_PI_2);
57 static const float kQuarterPi = static_cast<float>(0.5 * M_PI_2);
58 static const float kDegreesToRadians = static_cast<float>(M_PI / 180.0);
59 static const float kRadiansToDegrees = static_cast<float>(180.0 / M_PI);
60 static const float kMaxUniqueAngle = kPi;
61 static const float kDegreesPerCircle = 360.0f;

63 // The biggest floating point number > -pi.
64 // That is, there are no floats x s.t. -pi < x < kMinUniqueAngle.
65 //
66 // Useful for clamping to the valid angle range:
67 // [kMinUniqueAngle,kMaxUniqueAngle] == (-pi, pi]
68 // Note: "[" indicates an inclusive bound; "(" indicates exclusive.
69 //
70 // In general, after modding, we should clamp to the valid range.
71 // Floating point precision errors can generate numbers outside of
72 // (-pi, pi], even when the math is perfect.
73 static const float kMinUniqueAngle = -3.1415925f;

75 /// @class Angle
76 /// @brief Represent an angle in radians, uniquely in the range (-pi, pi].
77 ///
78 /// We include pi in the range, but exclude -pi, because pi and -pi
79 /// are equivalent mod 2pi.
80 ///
81 /// Equivalence is key to this class. We want only one representation
82 /// of every equivalent angle. For example, 0 and 2pi are both represented
83 /// as 0, internally. This unique representation allows for comparison and
84 /// precise arithmetic.
85 ///
86 /// All operators keep the angle values in the valid range.
87 ///
88 /// Why use instead of Quaternions?
89 /// -------------------------------
90 /// Quaternions are great for three dimensional rotations, but for many
91 /// applications you only have two dimensional rotations. Instead of the
92 /// four floats and heavy operations required by quaternions, an angle
93 /// can represent the rotation in one float and reasonably light operations.
94 /// Angles are easier to do trigonometry on. Also, they're conceptually
95 /// simpler.
96 ///
97 class Angle {
98  public:
99  Angle() : angle_(0.0f) {}

295  /// Wraps an angle to the range (-pi, pi].
296  /// This function is slow because it has a division. When possible, use
297  /// FromWithinThreePi instead.
298  static float WrapAngle(float angle) {
299  angle -= (floor(angle / kTwoPi) + 1.0f) * kTwoPi;
300  if (angle <= -kPi) {
301  angle += kTwoPi;
302  }
303  return angle;
304  }
305 
306  /// Create from `angle`, in radians, which is in the range (-3pi,3pi].
307  /// This function is significantly faster than WrapAngle since it avoids
308  /// division. It's also more precise for the same reason. The range may seem
309  /// strange at first glance; it's a consequence of the implementation. Just
310  /// know that any two sums of normalized angles will still be in the range
311  /// (-3pi,3pi].
312  static Angle FromWithinThreePi(const float angle) {
313  return Angle(ModWithinThreePi(angle));
314  }

394  /// Convert the angle into a 2D unit vector of the form
395  /// (`zero_axis`, `ninety_axis`).
396  /// When angle is 0 degrees, returns (1, 0), hence the name `zero_axis`.
397  /// When angle is 90 degrees, returns (0, 1), hence the name `ninety_axis`.
398  /// When angle is 180 degrees, returns (-1, 0).
399  /// When angle is -90 degrees, returns (0, -1).
400  void ToVector(float* const zero_axis, float* const ninety_axis) const {
401  // TODO OPT: Call single function that calculates both cos and sin.
402  *zero_axis = cos(angle_);
403  *ninety_axis = sin(angle_);

406  // Take 'angle' in the range (-3pi,3pi] and return an equivalent angle in
407  // the range (-pi,pi]. Note that angles are equivalent if they differ by 2pi.
408  static float ModWithinThreePi(const float angle) {
409  assert(-kThreePi < angle && angle < kThreePi);
410  // These ternary operators should be converted into select statements by
411  // the compiler.
412  const float above = angle < kMinUniqueAngle ? angle + kTwoPi : angle;
413  const float below = above > kMaxUniqueAngle ? above - kTwoPi : above;
414  assert(IsAngleInRange(below));
415  return below;
416  }
417 
418  static float ModIfNegativePi(const float angle) {
419  // Pi negates to -pi, which is outside the range so becomes +pi again.
420  return angle < kMinUniqueAngle ? kMaxUniqueAngle : angle;
421  }

470 inline Angle Angle::Clamp(const Angle& center, const Angle& max_diff) const {
471  assert(0 <= max_diff.angle_ && max_diff.angle_ <= kPi);
472 
473  // Get difference from 'center'. We know this will be a value in the range
474  // (-pi, pi].
475  const Angle diff = (*this) - center;
476 
477  // Clamp the difference to the valid range.
478  const Angle diff_clamped(
479  mathfu::Clamp(diff.angle_, -max_diff.angle_, max_diff.angle_));
480 
481  // Add the difference onto the center. Note that, if no clamping happened,
482  // we're left with *this.
483  return center + diff_clamped;
484 }
