31 static const float kPi = static_cast<float>(M_PI);
32 static const float kTwoPi = static_cast<float>(2.0 * M_PI);
33 static const float kThreePi = static_cast<float>(3.0 * M_PI);
34 static const float kHalfPi = static_cast<float>(M_PI_2);
35 static const float kDegreesToRadians = static_cast<float>(M_PI / 180.0);
36 static const float kRadiansToDegrees = static_cast<float>(180.0 / M_PI);
37 static const float kMaxUniqueAngle = kPi;
38 static const float kDegreesPerCircle = 360.0f;

40 // The biggest floating point number > -pi.
41 // That is, there are no floats x s.t. -pi < x < kMinUniqueAngle.
42 //
43 // Useful for clamping to the valid angle range:
44 // [kMinUniqueAngle,kMaxUniqueAngle] == (-pi, pi]
45 // Note: "[" indicates an inclusive bound; "(" indicates exclusive.
46 //
47 // In general, after modding, we should clamp to the valid range.
48 // Floating point precision errors can generate numbers outside of
49 // (-pi, pi], even when the math is perfect.
50 static const float kMinUniqueAngle = -3.1415925f;
51 
52 // Purpose
53 // =======
54 // Represents an angle in radians, uniquely in the range (-pi, pi].
55 //
56 // We include pi in the range, but exclude -pi, because pi and -pi
57 // are equivalent mod 2pi.
58 //
59 // Equivalence is key to this class. We want only one representation
60 // of every equivalent angle. For example, 0 and 2pi are both represented
61 // as 0, internally. This unique representation allows for comparison and
62 // precise arithmetic.
63 //
64 // All operators keep the angle values in the valid range.
65 //
66 // Conversions to and from vectors and matrices
67 // ============================================
68 // Conversions from and to XZ vectors follow the following convension:
69 // The vector is a point on unit circle corresponding to a sweep of 'angle'
70 // degrees from the x-axis towards the z-axis.
71 // 0 ==> ( 1, 0)
72 // pi/2 ==> ( 0, 1)
73 // pi ==> (-1, 0)
74 // 3pi/2 ==> ( 0, -1)
75 //
76 // Why use instead of Quaternions?
77 // ===============================
78 // Quaternions are great for three dimensional rotations, but for many
79 // applications you only have two dimensional rotations. Instead of the
80 // four floats and heavy operations required by quaternions, an angle
81 // can represent the rotation in one float and reasonably light operations.
82 // Angles are easier to do trigonometry on. Also, they're conceptually
83 // simpler.
84 //
85 class Angle {
86  public:
87  Angle() : angle_(0.0f) {}

147  // Wraps an angle to the range (-pi, pi].
148  static float WrapAngle(float angle) {
149  angle -= (floor(angle / kTwoPi) + 1.f) * kTwoPi;
150  if (angle <= -kPi) {
151  angle += kTwoPi;
152  }
153  return angle;
154  }
155 
156  // Create from 'angle', in radians, which is in the range (-3pi,3pi].
157  // This function is significantly faster than FromRadians since it avoids
158  // division. It's also more precise for the same reason. The range may seem
159  // strange at first glance; it's a consequence of the implementation. Just
160  // know that any two sums of normalized angles will still be in the range
161  // (-3pi,3pi].
162  static Angle FromWithinThreePi(const float angle) {
163  return Angle(ModWithinThreePi(angle));
164  }
165 
166  // Create from 'radians', which is converted to the range (-pi, pi].
167  static Angle FromRadians(const float radians) {
168  return Angle(WrapAngle(radians));
169  }
170 
171  // Create from 'degrees', which is converted to the range (-pi, pi].
172  static Angle FromDegrees(const float degrees) {
173  return FromRadians(degrees * kDegreesToRadians);
174  }

176  // Create from the x,z coordinates of a vector, as described in the
177  // comment at the head of the class.
178  static Angle FromXZVector(const mathfu::vec3& v) {
179  return Angle(ModIfNegativePi(atan2f(v[2], v[0])));
180  }

182  // Return true if 'angle' is within the valid range (-pi,pi], that is,
183  // the range inclusive of +pi but exclusive of -pi.
184  static bool IsAngleInRange(const float angle) {
185  return kMinUniqueAngle <= angle && angle <= kMaxUniqueAngle;
186  }

199  *x = cos(angle_);

203  // Take 'angle' in the range (-3pi,3pi] and return an equivalent angle in
204  // the range (-pi,pi]. Note that angles are equivalent if they differ by 2pi.
205  static float ModWithinThreePi(const float angle) {
206  assert(-kThreePi < angle && angle < kThreePi);
207  // These ternary operators should be converted into select statements by
208  // the compiler.
209  const float above = angle < kMinUniqueAngle ? angle + kTwoPi : angle;
210  const float below = above > kMaxUniqueAngle ? above - kTwoPi : above;
211  assert(IsAngleInRange(below));
212  return below;
213  }
214 
215  static float ModIfNegativePi(const float angle) {
216  // Pi negates to -pi, which is outside the range so becomes +pi again.
217  return angle < kMinUniqueAngle ? kMaxUniqueAngle : angle;
218  }

267 inline Angle Angle::Clamp(const Angle& center, const Angle& max_diff) const {
268  assert(0 <= max_diff.angle_ && max_diff.angle_ <= kPi);
269 
270  // Get difference from 'center'. We know this will be a value in the range
271  // (-pi, pi].
272  const Angle diff = (*this) - center;
273 
274  // Clamp the difference to the valid range.
275  const Angle diff_clamped(
276  mathfu::Clamp(diff.angle_, -max_diff.angle_, max_diff.angle_));
277 
278  // Add the difference onto the center. Note that, if no clamping happened,
279  // we're left with *this.
280  return center + diff_clamped;
281 }
