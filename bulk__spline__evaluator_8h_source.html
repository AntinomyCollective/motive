23 /// @class BulkSplineEvaluator
24 /// @brief Traverse through a set of splines in a performant way.
25 ///
26 /// This class should be used if you have hundreds or more splines that you need
27 /// to traverse in a uniform manner. It stores the spline data so that this
28 /// traversal is very fast, when done in bulk, and so we can take advantage of
29 /// SIMD on supported processors.
30 ///
31 /// This class maintains a current `x` value for each spline, and a current
32 /// cubic-curve for the segment of the spline corresponding to that `x`.
33 /// In AdvanceFrame, the `x`s are incremented. If this increment pushes us to
34 /// the next segment of a spline, the cubic-curve is reinitialized to the next
35 /// segment of the spline. The splines are evaluated at the current `x` in bulk.
36 ///
37 class BulkSplineEvaluator {
38  public:
39  typedef int Index;
40 
41  /// Increase or decrease the total number of indices processed.
42  ///
43  /// This class holds a set of splines, each is given an index
44  /// from 0 to size - 1.
45  ///
46  /// The number of splines can be increased or decreased with SetNumIndices().
47  /// - splines are allocated or removed at the highest indices
48  void SetNumIndices(const Index num_indices);
49 
50  /// Move the data at `old_index` into `new_index`.
51  ///
52  /// Unused indices are still processed every frame. You can fill these index
53  /// holes with MoveIndex(), to move items from the last index into the hole.
54  /// Once all holes have been moved to the highest indices, you can call
55  /// SetNumIndices() to stop processing these highest indices. Note that this
56  /// is exactly what fplutil::IndexAllocator does. You should use that class to
57  /// keep your indices contiguous.
58  void MoveIndex(const Index old_index, const Index new_index);
59 
60  /// Initialize `index` to clamp or wrap-around the `valid_y` range.
61  /// If `modular_arithmetic` is true, values wrap-around when they exceed
62  /// the bounds of `valid_y`. If it is false, values are clamped to `valid_y`.
63  void SetYRange(const Index index, const Range& valid_y,
64  const bool modular_arithmetic);
65 
66  /// Initialize `index` to process `s.spline` starting from `s.start_x`.
67  /// The Y() and Derivative() values are immediately available.
68  void SetSpline(const Index index, const CompactSpline& spline,
69  const SplinePlayback& playback);
70 
71  /// Mark spline as invalid.
72  void ClearSpline(const Index index) { sources_[index].spline = nullptr; }
73 
74  /// Reposition the spline at `index` evaluate from `x`.
75  /// Same as calling SetSpline() with the same spline and
76  /// `playback.start_x = x`.
77  void SetX(const Index index, const float x);
78 
79  /// Set conversion rate from AdvanceFrame's delta_x to the speed at which
80  /// we traverse the spline.
81  /// 0 ==> paused
82  /// 0.5 ==> half speed (slow motion)
83  /// 1 ==> authored speed
84  /// 2 ==> double speed (fast forward)
85  void SetPlaybackRate(const Index index, float playback_rate) {
86  playback_rates_[index] = playback_rate;
87  }
88 
89  /// Increment x and update the Y() and Derivative() values for all indices.
90  /// Process all indices in bulk to efficiently traverse memory and allow SIMD
91  /// instructions to be effective.
92  void AdvanceFrame(const float delta_x);
93 
94  /// Return true if the spline for `index` has valid spline data.
95  bool Valid(const Index index) const;
96 
97  /// Return the current x value for the spline at `index`.
98  float X(const Index index) const {
99  return CubicStartX(index) + cubic_xs_[index];
100  }
101 
102  /// Return the current y value for the spline at `index`.
103  float Y(const Index index) const { return ys_[index]; }
104 
105  /// Return the current y value for the spline at `index`, normalized to be
106  /// within the valid y_range.
107  float NormalizedY(const Index index) const {
108  return NormalizeY(index, ys_[index]);
109  }
110 
111  /// Return the current y value for spline indices
112  /// `index` ~ `index + count - 1`.
113  void Ys(const Index index, const Index count, float* ys) const {
114  assert(Valid(index) && Valid(index + count - 1));
115  memcpy(ys, &ys_[index], sizeof(ys[0]) * count);
116  }
117 
118  /// Return the current slope for the spline at `index`.
119  float Derivative(const Index index) const {
120  return PlaybackRate(index) * Cubic(index).Derivative(cubic_xs_[index]);
121  }
122 
123  /// Return the current slope for the spline at `index`. Ignore the playback
124  /// rate. This is useful for times when the playback rate is 0, but you
125  /// still want to get information about the underlying spline.
126  float DerivativeWithoutPlayback(const Index index) const {
127  return Cubic(index).Derivative(cubic_xs_[index]);
128  }
129 
130  /// Return the current playback rate of the spline at `index`.
131  float PlaybackRate(const Index index) const {
132  return playback_rates_[index];
133  }
134 
135  /// Return the spline that is currently being traversed at `index`.
136  const CompactSpline* SourceSpline(const Index index) const {
137  return sources_[index].spline;
138  }
139 
140  /// Return the raw cubic curve for `index`. Useful if you need to calculate
141  /// the second or third derivatives (which are not calculated in
142  /// AdvanceFrame), or plot the curve for debug reasons.
143  const CubicCurve& Cubic(const Index index) const { return cubics_[index]; }
144 
145  /// Return the current x value for the current cubic. Each spline segment
146  /// is evaluated as a cubic that starts at x=0.
147  float CubicX(const Index index) const { return cubic_xs_[index]; }
148 
149  /// Return x-value at the end of the spline.
150  float EndX(const Index index) const { return sources_[index].spline->EndX(); }
151 
152  /// Return y-value at the end of the spline.
153  float EndY(const Index index) const { return sources_[index].spline->EndY(); }
154 
155  /// Return slope at the end of the spline.
156  float EndDerivative(const Index index) const {
157  return PlaybackRate(index) * sources_[index].spline->EndDerivative();
158  }
159 
160  /// Return slope at the end of the spline, at `index`. Ignore the playback
161  /// rate. This is useful for times when the playback rate is 0, but you
162  /// still want to get information about the underlying spline.
163  float EndDerivativeWithoutPlayback(const Index index) const {
164  return sources_[index].spline->EndDerivative();
165  }
166 
167  /// Return y-distance between current-y and end-y.
168  /// If using modular arithmetic, consider both paths to the target
169  /// (directly and wrapping around), and return the length of the shorter path.
170  float YDifferenceToEnd(const Index index) const {
171  return NormalizeY(index, EndY(index) - Y(index));
172  }
173 
174  /// Apply modular arithmetic to ensure that `y` is within the valid y_range.
175  float NormalizeY(const Index index, const float y) const {
176  const YRange& r = y_ranges_[index];
177  return r.modular_arithmetic ? r.valid_y.NormalizeCloseValue(y) : y;
178  }
179 
180  /// Helper function to calculate the next y-value in a series of y-values
181  /// that are restricted by `direction`. There are always two paths that a y
182  /// value can take, in modular arithmetic. This function chooses the correct
183  /// one.
184  float NextY(const Index index, const float current_y, const float target_y,
185  const ModularDirection direction) const {
186  const YRange& r = y_ranges_[index];
187  if (!r.modular_arithmetic) return target_y;
188 
189  /// Calculate the difference from the current-y value for `direction`.
190  const float diff = r.valid_y.ModDiff(current_y, target_y, direction);
191  return current_y + diff;
192  }
193 
194  /// True if using modular arithmetic on this `index`. Modular arithmetic is
195  /// used for types such as angles, which are equivalent modulo 2pi
196  /// (e.g. -pi and +pi represent the same angle).
197  bool ModularArithmetic(const Index index) const {
198  return y_ranges_[index].modular_arithmetic != 0;
199  }
200 
201  /// The modular range for values that use ModularArithmetic(). Note that Y()
202  /// can be outside of this range. However, we always normalize to this range
203  /// before blending to a new spline.
204  const Range& ModularRange(const Index index) const {
205  return y_ranges_[index].valid_y;
206  }

235  struct Source {
236  /// Pointer to the source spline node. Spline data is owned externally.
237  /// We neither allocate or free this pointer here.
238  const CompactSpline* spline;
239 
240  /// Current index into `spline`. The cubics_ valid is instantiated from
241  /// spline[x_index].
242  CompactSplineIndex x_index;
243 
244  /// If true, start again at the beginning of the spline when we reach
245  /// the end.
246  bool repeat;
247 
248  Source()
249  : spline(nullptr), x_index(motive::kInvalidSplineIndex), repeat(false) {}
250  };
251 
252  struct YRange {
253  YRange() : valid_y(Range::Full()), modular_arithmetic(0) {}
254 
255  /// Hold the min and max extents of the modular range.
256  /// Modular ranges are used for things like angles, which wrap around from
257  /// -pi to +pi.
258  Range valid_y; // TODO: change name to modular_range.
259 
260  /// True if y values wrap around when they exit the valid_y range.
261  /// False if y values clamp to the edges of the valid_y range.
262  /// Use a mask for `modular_arithmetic` so that it can be used in `select`
263  /// instructions, in SIMD code.
264  uint32_t modular_arithmetic;
265  };
266 
267  // Data is organized in struct-of-arrays format to match the algorithm`s
268  // consumption of the data.
269  // - The algorithm that updates x values, and detects when we must transition
270  // to the next segment of the spline looks only at data in `cubic_xs_` and
271  // `cubic_x_ends_`.
272  // - The algorithm that updates `ys_` looks only at the data in `cubic_xs_`,
273  // `cubics_`, and `y_ranges_`. It writes to `ys_`.
274  // These vectors grow when SetNumIndices() is called, but they never shrink.
275  // So, we`ll have a few reallocs (which are slow) until the highwater mark is
276  // reached. Then the cost of reallocs disappears. In this way we have a
277  // reasonable tradeoff between memory conservation and runtime performance.
278 
279  /// Source spline nodes and our current index into these splines.
280  std::vector<Source> sources_;
281 
282  /// Define the valid output values. We can clamp to a range, or wrap around to
283  /// a range using modular arithmetic (two modes of operation).
284  std::vector<YRange> y_ranges_;
285 
286  /// The current `x` value at which `cubics_` are evaluated.
287  /// ys_[i] = cubics_[i].Evaluate(cubic_xs_[i])
288  std::vector<float> cubic_xs_;
289 
290  /// The last valid x value in `cubics_`.
291  std::vector<float> cubic_x_ends_;
292 
293  /// Speed at which time flows, relative to the spline's authored rate.
294  /// 0 ==> paused
295  /// 0.5 ==> half speed (slow motion)
296  /// 1 ==> authored speed
297  /// 2 ==> double speed (fast forward)
298  std::vector<float> playback_rates_;
299 
300  /// Currently active segment of sources_.spline.
301  /// Instantiated from
302  /// sources_[i].spline->CreateInitCubic(sources_[i].x_index).
303  std::vector<CubicCurve> cubics_;
304 
305  /// Value of the spline at `cubic_xs_`, normalized and clamped to be within
306  /// `y_ranges_`. Evaluated in AdvanceFrame.
307  std::vector<float> ys_;
308 
309  /// Stratch buffer used for internal calculations.
310  std::vector<Index> scratch_;
311 
312  /// Call the specified optimized functions, when available, instead of the
313  /// plain C++ functions. Note that we must perform this check at runtime,
314  /// not compile time: some platforms may or may not support all the
315  /// instructions used in the optimized functions. For example, some x86
316  /// processors support SSE3, others also support SSSE3, and still others
317  /// support neither. Therefore, x86 code always includes the C++ functions as
318  /// a fallback, and chooses the best functions at runtime.
319  ProcessorOptimization optimization_;
320 };
