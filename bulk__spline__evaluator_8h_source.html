23 // Traverse through a set of splines in a performant way.
24 //
25 // This class should be used if you have hundreds or more splines that you need
26 // to traverse in a uniform manner. It stores the spline data so that this
27 // traversal is very fast, when done in bulk, and so we can take advantage of
28 // SIMD on supported processors.
29 //
30 // This class maintains a current 'x' value for each spline, and a current
31 // cubic-curve for the segment of the spline corresponding to that 'x'.
32 // In AdvanceFrame, the 'x's are incremented. If this increment pushes us to
33 // the next segment of a spline, the cubic-curve is reinitialized to the next
34 // segment of the spline. The splines are evaluated at the current 'x' in bulk.
35 //
36 class BulkSplineEvaluator {
37  public:
38  typedef int Index;
39 
40  // This class holds a set of splines, each is given an index from 0 to size-1.
41  //
42  // The number of splines can be increased or decreased with SetNumIndices().
43  // - splines are allocated or removed at the highest indices
44  //
45  // Unused indices are still processed every frame. You can fill these index
46  // holes with MoveIndex(), to move items from the last index into the hole.
47  // Once all holes have been moved to the highest indices, you can call
48  // SetNumIndices() to stop processing these highest indices. Note that this
49  // is exactly what fpl::IndexAllocator does. You should use that class to
50  // keep your indices contiguous.
51  //
52  void SetNumIndices(const Index num_indices);
53  void MoveIndex(const Index old_index, const Index new_index);
54 
55  // Initialize 'index' to clamp or wrap-around the 'valid_y' range.
56  // If 'modular_arithmetic' is true, values wrap-around when they exceed
57  // the bounds of 'valid_y'. If it is false, values are clamped to 'valid_y'.
58  void SetYRange(const Index index, const Range& valid_y,
59  const bool modular_arithmetic);
60 
61  // Initialize 'index' to process 's.spline' starting from 's.start_x'.
62  // The Y() and Derivative() values are immediately available.
63  void SetSpline(const Index index, const SplinePlayback& s);
64 
65  // Increment x and update the Y() and Derivative() values for all indices.
66  // Process all indices in bulk to efficiently traverse memory and allow SIMD
67  // instructions to be effective.
68  void AdvanceFrame(const float delta_x);
69 
70  // Query the current state of the spline associated with 'index'.
71  bool Valid(const Index index) const;
72  float X(const Index index) const {
73  return CubicStartX(index) + cubic_xs_[index];
74  }
75  float Y(const Index index) const { return ys_[index]; }
76  float Derivative(const Index index) const {
77  return Cubic(index).Derivative(cubic_xs_[index]);
78  }
79  const CompactSpline* SourceSpline(const Index index) const {
80  return sources_[index].spline;
81  }
82 
83  // Get the raw cubic curve for 'index'. Useful if you need to calculate the
84  // second or third derivatives (which are not calculated in AdvanceFrame),
85  // or plot the curve for debug reasons.
86  const CubicCurve& Cubic(const Index index) const { return cubics_[index]; }
87  float CubicX(const Index index) const { return cubic_xs_[index]; }
88 
89  // Get state at end of the spline.
90  float EndX(const Index index) const { return sources_[index].spline->EndX(); }
91  float EndY(const Index index) const { return sources_[index].spline->EndY(); }
92  float EndDerivative(const Index index) const {
93  return sources_[index].spline->EndDerivative();
94  }
95 
96  // Return y-distance between current y and end-y. If using modular arithmetic,
97  // consider both paths to the target (directly and wrapping around),
98  // and return the length of the shorter path.
99  float YDifferenceToEnd(const Index index) const {
100  return NormalizeY(index, EndY(index) - Y(index));
101  }
102 
103  // Apply modular arithmetic to ensure that 'y' is within the valid y_range.
104  float NormalizeY(const Index index, const float y) const {
105  const YRange& r = y_ranges_[index];
106  return r.modular_arithmetic ? r.valid_y.Normalize(y) : y;
107  }
108 
109  // Helper function to calculate the next y-value in a series of y-values
110  // that are restricted by 'direction'. There are always two paths that a y
111  // value can take, in modular arithmetic. This function chooses the correct
112  // one.
113  float NextY(const Index index, const float current_y, const float target_y,
114  const ModularDirection direction) const {
115  const YRange& r = y_ranges_[index];
116  if (!r.modular_arithmetic) return target_y;
117 
118  // Calculate the difference from the current-y value for 'direction'.
119  const float diff = r.valid_y.ModDiff(current_y, target_y, direction);
120  return current_y + diff;
121  }
122 
123  private:
124  void InitCubic(const Index index, const float start_x);
125  Index NumIndices() const { return static_cast<Index>(sources_.size()); }
126  float SplineStartX(const Index index) const {
127  return sources_[index].spline->StartX();
128  }
129  float CubicStartX(const Index index) const {
130  const Source& s = sources_[index];
131  return s.spline->NodeX(s.x_index);
132  }
133 
134  // These functions have C and assembly language variants.
135  void UpdateCubicXsAndGetMask(const float delta_x, uint8_t* masks);
136  void UpdateCubicXsAndGetMask_C(const float delta_x, uint8_t* masks);
137  size_t UpdateCubicXs(const float delta_x, Index* indices_to_init);
138  size_t UpdateCubicXs_TwoSteps(const float delta_x, Index* indices_to_init);
139  size_t UpdateCubicXs_OneStep(const float delta_x, Index* indices_to_init);
140  void EvaluateIndex(const Index index);
141  void EvaluateCubics();
142  void EvaluateCubics_C();

144  struct Source {
145  // Pointer to the source spline node. Spline data is owned externally.
146  // We neither allocate or free this pointer here.
147  const CompactSpline* spline;
148 
149  // Current index into 'spline'. The cubics_ valid is instantiated from
150  // spline[x_index].
151  CompactSplineIndex x_index;
152 
153  // If true, start again at the beginning of the spline when we reach
154  // the end.
155  bool repeat;
156 
157  Source()
158  : spline(nullptr), x_index(fpl::kInvalidSplineIndex), repeat(false) {}
159  };
160 
161  struct YRange {
162  YRange() : valid_y(Range::Full()), modular_arithmetic(0) {}
163 
164  // Hold min and max values for the y result, or for the modular range.
165  // Modular ranges are used for things like angles, the wrap around from
166  // -pi to +pi.
167  Range valid_y;
168 
169  // True if y values wrap around when they exit the valid_y range.
170  // False if y values clamp to the edges of the valid_y range.
171  // Use a mask for 'modular_arithmetic' so that it can be used in 'select'
172  // instructions, in SIMD code.
173  uint32_t modular_arithmetic;
174  };
175 
176  // Data is organized in struct-of-arrays format to match the algorithm's
177  // consumption of the data.
178  // - The algorithm that updates x values, and detects when we must transition
179  // to the next segment of the spline looks only at data in 'cubic_xs_' and
180  // 'cubic_x_ends_'.
181  // - The algorithm that updates 'ys_' looks only at the data in 'cubic_xs_',
182  // 'cubics_', and 'y_ranges_'. It writes to 'ys_'.
183  // These vectors grow when SetNumIndices() is called, but they never shrink.
184  // So, we'll have a few reallocs (which are slow) until the highwater mark is
185  // reached. Then the cost of reallocs disappears. In this way we have a
186  // reasonable tradeoff between memory conservation and runtime performance.
187 
188  // Source spline nodes and our current index into these splines.
189  std::vector<Source> sources_;
190 
191  // Define the valid output values. We can clamp to a range, or wrap around to
192  // a range using modular arithmetic (two modes of operation).
193  std::vector<YRange> y_ranges_;
194 
195  // The current 'x' value at which 'cubics_' are evaluated.
196  // ys_[i] = cubics_[i].Evaluate(cubic_xs_[i])
197  std::vector<float> cubic_xs_;
198 
199  // The last valid x value in 'cubics_'.
200  std::vector<float> cubic_x_ends_;
201 
202  // Currently active segment of sources_.spline.
203  // Instantiated from sources_[i].spline->CreateInitCubic(sources_[i].x_index).
204  std::vector<CubicCurve> cubics_;
205 
206  // Value of the spline at 'cubic_xs_', normalized and clamped to be within
207  // 'y_ranges_'. Evaluated in AdvanceFrame.
208  std::vector<float> ys_;
209 
210  // Stratch buffer used for internal calculations.
211  std::vector<Index> scratch_;
212 
213  // Call the specified optimized functions, when available, instead of the
214  // plain C++ functions. Note that we must perform this check at runtime,
215  // not compile time: some platforms may or may not support all the
216  // instructions used in the optimized functions. For example, some x86
217  // processors support SSE3, others also support SSSE3, and still others
218  // support neither. Therefore, x86 code always includes the C++ functions as
219  // a fallback, and chooses the best functions at runtime.
220  ProcessorOptimization optimization_;
221 };
