78 /// @class ModularInit
79 /// Base-class for OvershootInit and SmoothInit. Holds parameters related
80 /// to modular arithmetic in a Motivator.
81 class ModularInit : public MotivatorInit {
82  public:
83  explicit ModularInit(MotivatorType type)
84  : MotivatorInit(type), range_(Range::Full()), modular_(false) {}
85 
86  /// The derived type must call this constructor with it's MotivatorType
87  /// identifier.
88  /// @param type The kType parameter of the derived init class.
89  /// @param range The range we should clamp Motivator::Value() to. If you
90  /// don't want to clamp, then leave as Range::Full().
91  /// @param modular Option to use modular arithmetic for Motivator::Value().
92  /// If true, all values are wrapped around to stay within
93  /// `range`.
94  ModularInit(MotivatorType type, const Range& range, bool modular)
95  : MotivatorInit(type), range_(range), modular_(modular) {}
96 
97  /// Ensure position `x` is within the valid constraint range.
98  /// `x` must be within range_.Length() of range_. This is a reasonable
99  /// restriction in most cases (such as after an arithmetic operation).
100  /// For cases where `x` may be wildly outside the range, use
101  /// NormalizeWildValue() instead.
102  /// @param x The value to be modded to within `range_`
103  float Normalize(float x) const { return modular_ ? range_.Normalize(x) : x; }
104 
105  /// Normalize `x` value well outside the range. Significantly slower than
106  /// the simple Normalize().
107  /// @param x The value to be modded to within `range_`
108  float NormalizeWildValue(float x) const {
109  return modular_ ? range_.NormalizeWildValue(x) : x;
110  }
111 
112  /// Ensure the motivator value is within the specified range.
113  /// @param x The value to clamped to `range_`
114  float ClampValue(float x) const { return range_.Clamp(x); }
115 
116  /// Return minimum value of the range.
117  float Min() const { return range_.start(); }
118 
119  /// Return maximum value of the range.
120  float Max() const { return range_.end(); }
121 
122  const Range& range() const { return range_; }
123  void set_range(const Range& r) { range_ = r; }
124 
125  bool modular() const { return modular_; }
126  void set_modular(bool modular) { modular_ = modular; }
127 
128  private:
129  /// Minimum and maximum values for Motivator::Value().
130  /// Clamp (if modular_ is false) or wrap-around (if modular_ is true) when
131  /// we reach these boundaries.
132  Range range_;
133 
134  /// A modular value wraps around from min to max. For example, an angle
135  /// is modular, where -pi is equivalent to +pi. Setting this to true ensures
136  /// that arithmetic wraps around instead of clamping to min/max.
137  bool modular_;
138 };

141  public:
142  MOTIVE_INTERFACE();
143 
144  OvershootInit()
145  : ModularInit(kType),
146  max_velocity_(0.0f),
147  accel_per_difference_(0.0f),
148  wrong_direction_multiplier_(0.0f),
149  max_delta_time_(0) {}
150 
151  /// Ensure velocity is within the reasonable limits.
152  float ClampVelocity(float velocity) const {
153  return mathfu::Clamp(velocity, -max_velocity_, max_velocity_);
154  }
155 
156  /// Ensure the Motivator's 'value' doesn't increment by more than 'max_delta'.
157  /// This is different from ClampVelocity because it is independent of time.
158  /// No matter how big the timestep, the delta will not be too great.
159  float ClampDelta(float delta) const {
160  return mathfu::Clamp(delta, -max_delta_, max_delta_);
161  }
162 
163  /// Return true if we're close to the target and almost stopped.
164  /// The definition of "close to" and "almost stopped" are given by the
165  /// "at_target" member.
166  bool AtTarget(float dist, float velocity) const {
167  return at_target_.Settled(dist, velocity);
168  }
169 
170  float max_velocity() const { return max_velocity_; }
171  float max_delta() const { return max_delta_; }
172  const Settled1f& at_target() const { return at_target_; }
173  Settled1f& at_target() { return at_target_; }
174  float accel_per_difference() const { return accel_per_difference_; }
175  float wrong_direction_multiplier() const {
176  return wrong_direction_multiplier_;
177  }
178  MotiveTime max_delta_time() const { return max_delta_time_; }
179 
180  void set_max_velocity(float max_velocity) { max_velocity_ = max_velocity; }
181  void set_max_delta(float max_delta) { max_delta_ = max_delta; }
182  void set_at_target(const Settled1f& at_target) { at_target_ = at_target; }
183  void set_accel_per_difference(float accel_per_difference) {
184  accel_per_difference_ = accel_per_difference;
185  }
186  void set_wrong_direction_multiplier(float wrong_direction_multiplier) {
187  wrong_direction_multiplier_ = wrong_direction_multiplier;
188  }
189  void set_max_delta_time(MotiveTime max_delta_time) {
190  max_delta_time_ = max_delta_time;
191  }
192 
193  private:
194  /// Maximum speed at which the value can change. That is, maximum value for
195  /// the Motivator::Velocity(). In units/tick.
196  /// For example, if the value is an angle, then this is the max angular
197  /// velocity, and the units are radians/tick.
198  float max_velocity_;
199 
200  /// Maximum that Motivator::Value() can be altered on a single call to
201  /// MotiveEngine::AdvanceFrame(), regardless of velocity or delta_time.
202  float max_delta_;
203 
204  /// Cutoff to determine if the Motivator's current state has settled on the
205  /// target. Once it has settled, Value() is set to TargetValue() and
206  /// Velocity() is set to zero.
207  Settled1f at_target_;
208 
209  /// Acceleration is a multiple of abs('state_.position' - 'target_.position').
210  /// Bigger differences cause faster acceleration.
211  float accel_per_difference_;
212 
213  /// When accelerating away from the target, we multiply our acceleration by
214  /// this amount. We need counter-acceleration to be stronger so that the
215  /// amplitude eventually dies down; otherwise, we'd just have a pendulum.
216  float wrong_direction_multiplier_;
217 
218  /// The algorithm is iterative. When the iteration step gets too big, the
219  /// behavior becomes erratic. This value clamps the iteration step.
220  MotiveTime max_delta_time_;
221 };

286 /// @class MatrixInit
287 /// @brief Initialize a MatrixMotivator4f to generate its matrix from
288 /// a series of operations.
289 ///
290 /// Initialize a MatrixMotivator4f with these initialization parameters to
291 /// create a motivator that generates a 4x4 matrix from a series of basic
292 /// matrix operations. The basic matrix operations are driven by one dimensional
293 /// motivators.
294 ///
295 /// The series of operations can transform an object from the coordinate space
296 /// in which it was authored, to world (or local) space. For example, if you
297 /// have a penguin that is authored at (0,0,0) facing up the x-axis, you can
298 /// move it to it's target position with four operations:
299 ///
300 /// kScaleUniformly --> to make penguin the correct size
301 /// kRotateAboutY --> to make penguin face the correct direction
302 /// kTranslateX } --> to move penguin along to ground to target position
303 /// kTranslateZ }
304 class MatrixOpArray {
305  public:
306  typedef std::vector<MatrixOperationInit> OpVector;
307 
308  // Guess at the number of operations we'll have. Better to high-ball a little
309  // so that we don't have to reallocate the `ops_` vector.
310  static const int kDefaultExpectedNumOps = 8;
311 
312  /// By default expect a relatively high number of ops. Cost for allocating
313  /// a bit too much temporary memory is small compared to cost of reallocating
314  /// that memory.
315  explicit MatrixOpArray(int expected_num_ops = kDefaultExpectedNumOps) {
316  ops_.reserve(expected_num_ops);
317  }
318 
319  /// Remove all matrix operations from the sequence.
320  void Clear(int expected_num_ops = kDefaultExpectedNumOps) {
321  ops_.clear();
322  ops_.reserve(expected_num_ops);
323  }
324 
325  /// Operation is constant. For example, use to put something flat on the
326  /// ground, with 'type' = kRotateAboutX and 'const_value' = pi/2.
327  void AddOp(MatrixOperationType type, float const_value) {
328  ops_.push_back(MatrixOperationInit(type, const_value));
329  }
330 
331  /// Operation is driven by a one dimensional motivator. For example, you can
332  /// control the face angle of a standing object with 'type' = kRotateAboutY
333  /// and 'init' a curve specified by SmoothInit.
334  void AddOp(MatrixOperationType type, const MotivatorInit& init) {
335  ops_.push_back(MatrixOperationInit(type, init));
336  }
337 
338  /// Operation is driven by a one dimensional motivator, and initial value
339  /// is specified.
340  void AddOp(MatrixOperationType type, const MotivatorInit& init,
341  float initial_value) {
342  ops_.push_back(MatrixOperationInit(type, init, initial_value));
343  }
344 
345  /// Operation is driven by a one dimensional motivator, which is initialized
346  /// to traverse the key points specified in `target`.
347  void AddOp(MatrixOperationType type, const MotivatorInit& init,
348  const MotiveTarget1f& target) {
349  ops_.push_back(MatrixOperationInit(type, init, target));
350  }
351 
352  /// Operation is driven by a one dimensional motivator, which is initialized
353  /// to follow the predefined curve specified in `spline`.
354  void AddOp(MatrixOperationType type, const MotivatorInit& init,
355  const CompactSpline& spline) {
356  ops_.push_back(MatrixOperationInit(type, init, spline));
357  }
358 
359  // Maximum duration of any of the splines.
360  MotiveTime EndTime() const {
361  MotiveTime end_time = 0;
362  for (size_t i = 0; i < ops_.size(); ++i) {
363  const MatrixOperationInit& op = ops_[i];
364  if (op.union_type == MatrixOperationInit::kUnionSpline) {
365  end_time =
366  std::max(end_time, static_cast<MotiveTime>(op.spline->EndX()));
367  }
368  }
369  return end_time;
370  }
371 
372  const OpVector& ops() const { return ops_; }
373 
374  private:
375  OpVector ops_;
376 };
