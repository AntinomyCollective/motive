37 /// @class ModularInit
38 /// Base-class for OvershootInit and SmoothInit. Holds parameters related
39 /// to modular arithmetic in a Motivator.
40 class ModularInit : public MotivatorInit {
41  public:
42  /// The derived type must call this constructor with it's MotivatorType
43  /// identifier.
44  /// @param type The kType parameter of the derived init class.
45  explicit ModularInit(MotivatorType type)
46  : MotivatorInit(type), range_(fpl::Range::Full()), modular_(false) {}
47  ModularInit(MotivatorType type, const fpl::Range& range, bool modular)
48  : MotivatorInit(type), range_(range), modular_(modular) {}
49 
50  /// Ensure position `x` is within the valid constraint range.
51  /// `x` must be within range_.Length() of range_. This is a reasonable
52  /// restriction in most cases (such as after an arithmetic operation).
53  /// For cases where `x` may be wildly outside the range, use
54  /// NormalizeWildValue() instead.
55  /// @param x The value to be modded to within `range_`
56  float Normalize(float x) const { return modular_ ? range_.Normalize(x) : x; }
57 
58  /// Normalize `x` value well outside the range. Significantly slower than
59  /// the simple Normalize().
60  /// @param x The value to be modded to within `range_`
61  float NormalizeWildValue(float x) const {
62  return modular_ ? range_.NormalizeWildValue(x) : x;
63  }
64 
65  /// Ensure the motivator value is within the specified range.
66  /// @param x The value to clamped to `range_`
67  float ClampValue(float x) const { return range_.Clamp(x); }
68 
69  /// Return minimum value of the range.
70  float Min() const { return range_.start(); }
71 
72  /// Return maximum value of the range.
73  float Max() const { return range_.end(); }
74 
75  const fpl::Range& range() const { return range_; }
76  void set_range(const fpl::Range& r) { range_ = r; }
77 
78  bool modular() const { return modular_; }
79  void set_modular(bool modular) { modular_ = modular; }
80 
81  private:
82  /// Minimum and maximum values for Motivator::Value().
83  /// Clamp (if modular_ is false) or wrap-around (if modular_ is true) when
84  /// we reach these boundaries.
85  fpl::Range range_;
86 
87  /// A modular value wraps around from min to max. For example, an angle
88  /// is modular, where -pi is equivalent to +pi. Setting this to true ensures
89  /// that arithmetic wraps around instead of clamping to min/max.
90  bool modular_;
91 };

93 class OvershootInit : public ModularInit {
94  public:
95  MOTIVE_INTERFACE();
96 
97  OvershootInit()
98  : ModularInit(kType),
99  max_velocity_(0.0f),
100  accel_per_difference_(0.0f),
101  wrong_direction_multiplier_(0.0f),
102  max_delta_time_(0) {}
103 
104  /// Ensure velocity is within the reasonable limits.
105  float ClampVelocity(float velocity) const {
106  return mathfu::Clamp(velocity, -max_velocity_, max_velocity_);
107  }
108 
109  /// Ensure the Motivator's 'value' doesn't increment by more than 'max_delta'.
110  /// This is different from ClampVelocity because it is independent of time.
111  /// No matter how big the timestep, the delta will not be too great.
112  float ClampDelta(float delta) const {
113  return mathfu::Clamp(delta, -max_delta_, max_delta_);
114  }
115 
116  /// Return true if we're close to the target and almost stopped.
117  /// The definition of "close to" and "almost stopped" are given by the
118  /// "at_target" member.
119  bool AtTarget(float dist, float velocity) const {
120  return at_target_.Settled(dist, velocity);
121  }
122 
123  float max_velocity() const { return max_velocity_; }
124  float max_delta() const { return max_delta_; }
125  const Settled1f& at_target() const { return at_target_; }
126  Settled1f& at_target() { return at_target_; }
127  float accel_per_difference() const { return accel_per_difference_; }
128  float wrong_direction_multiplier() const {
129  return wrong_direction_multiplier_;
130  }
131  MotiveTime max_delta_time() const { return max_delta_time_; }
132 
133  void set_max_velocity(float max_velocity) { max_velocity_ = max_velocity; }
134  void set_max_delta(float max_delta) { max_delta_ = max_delta; }
135  void set_at_target(const Settled1f& at_target) { at_target_ = at_target; }
136  void set_accel_per_difference(float accel_per_difference) {
137  accel_per_difference_ = accel_per_difference;
138  }
139  void set_wrong_direction_multiplier(float wrong_direction_multiplier) {
140  wrong_direction_multiplier_ = wrong_direction_multiplier;
141  }
142  void set_max_delta_time(MotiveTime max_delta_time) {
143  max_delta_time_ = max_delta_time;
144  }
145 
146  private:
147  /// Maximum speed at which the value can change. That is, maximum value for
148  /// the Motivator::Velocity(). In units/tick.
149  /// For example, if the value is an angle, then this is the max angular
150  /// velocity, and the units are radians/tick.
151  float max_velocity_;
152 
153  /// Maximum that Motivator::Value() can be altered on a single call to
154  /// MotiveEngine::AdvanceFrame(), regardless of velocity or delta_time.
155  float max_delta_;
156 
157  /// Cutoff to determine if the Motivator's current state has settled on the
158  /// target. Once it has settled, Value() is set to TargetValue() and
159  /// Velocity() is set to zero.
160  Settled1f at_target_;
161 
162  /// Acceleration is a multiple of abs('state_.position' - 'target_.position').
163  /// Bigger differences cause faster acceleration.
164  float accel_per_difference_;
165 
166  /// When accelerating away from the target, we multiply our acceleration by
167  /// this amount. We need counter-acceleration to be stronger so that the
168  /// amplitude eventually dies down; otherwise, we'd just have a pendulum.
169  float wrong_direction_multiplier_;
170 
171  /// The algorithm is iterative. When the iteration step gets too big, the
172  /// behavior becomes erratic. This value clamps the iteration step.
173  MotiveTime max_delta_time_;
174 };

193 struct MatrixOperationInit {
194  enum UnionType {
195  kUnionEmpty,
196  kUnionInitialValue,
197  kUnionTarget,
198  kUnionSpline
199  };
200 
201  /// Matrix operation never changes. Always use 'const_value'.
202  MatrixOperationInit(MatrixOperationType type, float const_value)
203  : init(nullptr),
204  type(type),
205  union_type(kUnionInitialValue),
206  initial_value(const_value) {}
207 
208  /// Matrix operation is driven by Motivator defined by 'init'.
209  MatrixOperationInit(MatrixOperationType type, const MotivatorInit& init)
210  : init(&init), type(type), union_type(kUnionEmpty) {}
211 
212  /// Matrix operation is driven by Motivator defined by 'init'. Specify initial
213  /// value as well.
214  MatrixOperationInit(MatrixOperationType type, const MotivatorInit& init,
215  float initial_value)
216  : init(&init),
217  type(type),
218  union_type(kUnionInitialValue),
219  initial_value(initial_value) {}
220 
221  MatrixOperationInit(MatrixOperationType type, const MotivatorInit& init,
222  const MotiveTarget1f& target)
223  : init(&init), type(type), union_type(kUnionTarget), target(&target) {}
224 
225  MatrixOperationInit(MatrixOperationType type, const MotivatorInit& init,
226  const fpl::SplinePlayback& spline)
227  : init(&init), type(type), union_type(kUnionSpline), spline(&spline) {}
228 
229  const MotivatorInit* init;
230  MatrixOperationType type;
231  UnionType union_type;
232  union {
233  float initial_value;
234  const MotiveTarget1f* target;
235  const fpl::SplinePlayback* spline;
236  };

239 /// @class MatrixInit
240 /// @brief Initialize a MotivatorMatrix4f to generate its matrix from
241 /// a series of operations.
242 ///
243 /// Initialize an MotivatorMatrix4f with these initialization parameters to
244 /// create an motivator that generates a 4x4 matrix from a series of basic
245 /// matrix operations. The basic matrix operations are driven by 1 dimensional
246 /// motivators.
247 ///
248 /// The series of operations can transform an object from the coordinate space
249 /// in which it was authored, to world (or local) space. For example, if you
250 /// have a penguin that is authored at (0,0,0) facing up the x-axis, you can
251 /// move it to it's target position with four operations:
252 ///
253 /// kScaleUniformly --> to make penguin the correct size
254 /// kRotateAboutY --> to make penguin face the correct direction
255 /// kTranslateX } --> to move penguin along to ground to target position
256 /// kTranslateZ }
257 class MatrixInit : public MotivatorInit {
258  public:
259  MOTIVE_INTERFACE();
260  typedef std::vector<MatrixOperationInit> OpVector;
261 
262  /// By default expect a relatively high number of ops. Cost for allocating
263  /// a bit too much temporary memory is small compared to cost of reallocating
264  /// that memory.
265  explicit MatrixInit(int expected_num_ops = 8) : MotivatorInit(kType) {
266  ops_.reserve(expected_num_ops);
267  }
268 
269  /// Remove all matrix operations from the sequence.
270  void Clear() { ops_.clear(); }
271 
272  /// Operation is constant. For example, use to put something flat on the
273  /// ground, with 'type' = kRotateAboutX and 'const_value' = pi/2.
274  void AddOp(MatrixOperationType type, float const_value) {
275  ops_.push_back(MatrixOperationInit(type, const_value));
276  }
277 
278  /// Operation is driven by a 1-dimensional motivator. For example, you can
279  /// control the face angle of a standing object with 'type' = kRotateAboutY
280  /// and 'init' a curve specified by SmoothInit.
281  void AddOp(MatrixOperationType type, const MotivatorInit& init) {
282  ops_.push_back(MatrixOperationInit(type, init));
283  }
284 
285  /// Operation is driven by a 1-dimensional motivator, and initial value
286  /// is specified.
287  void AddOp(MatrixOperationType type, const MotivatorInit& init,
288  float initial_value) {
289  ops_.push_back(MatrixOperationInit(type, init, initial_value));
290  }
291 
292  /// Operation is driven by a 1-dimensional motivator, which is initialized
293  /// to traverse the key points specified in `target`.
294  void AddOp(MatrixOperationType type, const MotivatorInit& init,
295  const MotiveTarget1f& target) {
296  ops_.push_back(MatrixOperationInit(type, init, target));
297  }
298 
299  /// Operation is driven by a 1-dimensional motivator, which is initialized
300  /// to follow the predefined curve specified in `spline`.
301  void AddOp(MatrixOperationType type, const MotivatorInit& init,
302  const fpl::SplinePlayback& spline) {
303  ops_.push_back(MatrixOperationInit(type, init, spline));
304  }
305 
306  const OpVector& ops() const { return ops_; }
307 
308  private:
309  OpVector ops_;
310 };
