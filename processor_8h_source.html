34 // MotiveProcessor
35 // ==============
36 // An MotiveProcessor processes *all* instances of one type of Motivator.
37 // Or, at least, all instances within a given MotiveEngine.
38 //
39 // We pool the processing for potential optimization opportunities. We may have
40 // hundreds of smoothly-interpolating one-dimensional Motivators, for example.
41 // It's nice to be able to update those 4 or 8 or 16 at a time using SIMD.
42 // And it's nice to have the data gathered in one spot if we want to use
43 // multiple threads.
44 //
45 // MotiveProcessors exists in the internal API. For the external API, please see
46 // Motivator.
47 //
48 // Users can create their own Motivator algorithms by deriving from
49 // MotiveProcessor. MotiveProcessors must have a factory that's registered with
50 // the MotiveEngine (please see MotiveEngine for details). Once registered,
51 // you can use your new Motivator algorithm by calling Motivator::Initialize()
52 // with Init::type set to your MotiveProcessor's MotivatorType.
53 //
54 // MotiveProcessors run on mathfu types. Please see the specializations below
55 // for MotiveProcessors of various dimensions. Note that Motivator expects the
56 // MotiveProcessor to return a mathfu type, so you shouldn't try to specialize
57 // MotiveProcessor<> with any of your native types.
58 //
59 class MotiveProcessor {
60  public:
61  MotiveProcessor()
62  : allocator_callbacks_(this), index_allocator_(allocator_callbacks_) {}
63  virtual ~MotiveProcessor();
64 
65  // Instantiate motivator data inside the MotiveProcessor, and initialize
66  // 'motivator' as a reference to that data.
67  // The 'engine' is required if the MotiveProcessor itself creates child
68  // Motivators. This function should only be called by Motivator::Initialize().
69  void InitializeMotivator(const MotivatorInit& init, MotiveEngine* engine,
70  Motivator* motivator);
71 
72  // Remove an motivator and return its index to the pile of allocatable
73  // indices.
74  // Should only be called by Motivator::Invalidate().
75  void RemoveMotivator(MotiveIndex index);
76 
77  // Transfer ownership of the motivator at 'index' to 'new_motivator'.
78  // Resets the Motivator that currently owns 'index' and initializes
79  // 'new_motivator'.
80  // Should only be called by Motivator copy operations.
81  void TransferMotivator(MotiveIndex index, Motivator* new_motivator);
82 
83  // Returns true if 'index' is currently driving an motivator.
84  bool ValidIndex(MotiveIndex index) const;
85 
86  // Returns true if 'index' is currently driving 'motivator'.
87  bool ValidMotivator(MotiveIndex index, const Motivator* motivator) const {
88  return ValidIndex(index) && motivators_[index] == motivator;
89  }
90 
91  // Advance the simulation by delta_time.
92  // Should only be called by MotiveEngine::AdvanceFrame.
93  virtual void AdvanceFrame(MotiveTime delta_time) = 0;
94 
95  // Return GUID representing the Motivator's type. Must be implemented by
96  // derived class.
97  virtual MotivatorType Type() const = 0;
98 
99  // The number of floats (or doubles) being animated. For example, a position
100  // in 3D space would return 3.
101  virtual int Dimensions() const = 0;
102 
103  // The lower the number, the sooner the MotiveProcessor gets updated.
104  // Should never change. We want a static ordering of processors.
105  // Some MotiveProcessors use the output of other MotiveProcessors, so
106  // we impose a strict ordering here.
107  virtual int Priority() const = 0;
108 
109  // Ensure that the internal state is consistent. Call periodically when
110  // debugging problems where the internal state is corrupt.
111  void VerifyInternalState() const;
112 
113  protected:
114  // Initialize data at 'index'. The meaning of 'index' is determined by the
115  // MotiveProcessor implementation (most likely it is the index into one or
116  // more data_ arrays though).
117  // MotiveProcessor tries to keep the 'index' as low as possible, by
118  // recycling ones that have been freed, and by providing a Defragment()
119  // function to move later indices to indices that have been freed.
120  virtual void InitializeIndex(const MotivatorInit& init, MotiveIndex index,
121  MotiveEngine* engine) = 0;
122 
123  // Reset data at 'index'. See comment above InitializeIndex for meaning of
124  // 'index'. If your MotiveProcessor stores data in a plain array, you
125  // probably have nothing to do. But if you use dynamic memory per index,
126  // (which you really shouldn't - too slow!), you should deallocate it here.
127  // For debugging, it might be nice to invalidate the data.
128  virtual void RemoveIndex(MotiveIndex index) = 0;
129 
130  // Move the data at 'old_index' into 'new_index'. Used by Defragment().
131  // Note that 'new_index' is guaranteed to be inactive.
132  virtual void MoveIndex(MotiveIndex old_index, MotiveIndex new_index) = 0;
133 
134  virtual void SetNumIndices(MotiveIndex num_indices) = 0;
135 
136  void MoveIndexBase(MotiveIndex old_index, MotiveIndex new_index);
137  void SetNumIndicesBase(MotiveIndex num_indices);
138 
139  // When an index is moved, the Motivator that references that index is
140  // updated.
141  // Can be called at the discretion of your MotiveProcessor, but normally
142  // called
143  // at the beginning of your MotiveProcessor::AdvanceFrame.
144  void Defragment() { index_allocator_.Defragment(); }
145 
146  private:
147  // Don't notify derived class.
148  void RemoveMotivatorWithoutNotifying(MotiveIndex index);
149 
150  // Proxy callbacks from IndexAllocator into MotiveProcessor.
151  class AllocatorCallbacks
152  : public fpl::IndexAllocator<MotiveIndex>::CallbackInterface {
153  public:
154  AllocatorCallbacks(MotiveProcessor* processor) : processor_(processor) {}
155  virtual void SetNumIndices(MotiveIndex num_indices) {
156  processor_->SetNumIndicesBase(num_indices);
157  }
158  virtual void MoveIndex(MotiveIndex old_index, MotiveIndex new_index) {
159  processor_->MoveIndexBase(old_index, new_index);
160  }
161 
162  private:
163  MotiveProcessor* processor_;
164  };
165 
166  // Back-pointer to the Motivators for each index. The Motivators reference
167  // this
168  // MotiveProcessor and a specific index into the MotiveProcessor, so when the
169  // index is moved, or when the MotiveProcessor itself is destroyed, we need
170  // to update the Motivator.
171  // Note that we only keep a reference to a single Motivator per index. When
172  // a copy of an Motivator is made, the old Motivator is Reset and the
173  // reference
174  // here is updated.
175  std::vector<Motivator*> motivators_;
176 
177  // Proxy calbacks into MotiveProcessor. The other option is to derive
178  // MotiveProcessor from IndexAllocator::CallbackInterface, but that would
179  // create a messier API, and not be great OOP.
180  // This member should be initialized before index_allocator_ is initialized.
181  AllocatorCallbacks allocator_callbacks_;
182 
183  // When an index is freed, we keep track of it here. When an index is
184  // allocated, we use one off this array, if one exists.
185  // When Defragment() is called, we empty this array by filling all the
186  // unused indices with the highest allocated indices. This reduces the total
187  // size of the data arrays.
188  fpl::IndexAllocator<MotiveIndex> index_allocator_;
189 };

191 // Interface for motivator types that drive a single float value.
192 // That is, for MotiveProcessors that interface with Motivator1f's.
193 class MotiveProcessor1f : public MotiveProcessor {
194  public:
195  virtual int Dimensions() const { return 1; }
196 
197  // Get current motivator values from the processor.
198  virtual float Value(MotiveIndex index) const = 0;
199  virtual float Velocity(MotiveIndex index) const = 0;
200  virtual float TargetValue(MotiveIndex index) const = 0;
201  virtual float TargetVelocity(MotiveIndex index) const = 0;
202  virtual float Difference(MotiveIndex index) const = 0;
203  virtual MotiveTime TargetTime(MotiveIndex index) const = 0;
204 
205  // At least one of these should be implemented. Otherwise, there will be
206  // no way to drive the Motivator towards a target.
207  virtual void SetTarget(MotiveIndex /*index*/, const MotiveTarget1f& /*t*/) {}

209  const fpl::SplinePlayback& /*s*/) {}
210 };

212 // Interface for motivator types that drive a 4x4 float matrix.
213 // That is, for MotiveProcessors that interface with MotivatorMatrix4f's.
214 class MotiveProcessorMatrix4f : public MotiveProcessor {
215  public:
216  virtual int Dimensions() const { return 16; }
217 
218  // Get the current matrix value from the processor.
219  virtual const mathfu::mat4& Value(MotiveIndex index) const = 0;
220 
221  // Get current values of the components that create the matrix.
222  virtual float ChildValue1f(MotiveIndex index,
223  MotiveChildIndex child_index) const = 0;
224  virtual mathfu::vec3 ChildValue3f(MotiveIndex index,
225  MotiveChildIndex child_index) const {
226  return mathfu::vec3(ChildValue1f(index, child_index),
227  ChildValue1f(index, child_index + 1),
228  ChildValue1f(index, child_index + 2));
229  }
230 
231  // Set child values. Matrices are composed from child components.
232  virtual void SetChildTarget1f(MotiveIndex /*index*/,
233  MotiveChildIndex /*child_index*/,
234  const MotiveTarget1f& /*t*/) {}
235  virtual void SetChildValue1f(MotiveIndex /*index*/,
236  MotiveChildIndex /*child_index*/,
237  float /*value*/) {}
238  virtual void SetChildValue3f(MotiveIndex index, MotiveChildIndex child_index,
239  const mathfu::vec3& value) {
240  for (int i = 0; i < 3; ++i) {
241  SetChildValue1f(index, child_index + i, value[i]);
242  }
243  }
244 };
