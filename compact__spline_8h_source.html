25 /// @typedef CompactSplineXGrain
26 /// X-axis is quantized into units of `x_granularity`. X values are represented
27 /// by multiples of `x_granularity`. One unit of CompactSplineXGrain represents
28 /// one multiple of `x_granularity`.

31 /// @typedef CompactSplineYRung
32 /// Y values within `y_range` can be represented. We quantize the `y_range`
33 /// into equally-sized rungs, and round to the closest rung.

62 /// @class CompactSpline
63 /// @brief Represent a smooth curve in a small amount of memory.
64 ///
65 /// This spline interpolates a series of (x, y, derivative) nodes to create a
66 /// smooth curve.
67 ///
68 /// This class holds a series of such nodes, and aids with the construction of
69 /// that series by inserting extra nodes when extra smoothness is required.
70 ///
71 /// The data in this class is compacted as quantized values. It's not intended
72 /// to be read directly. You should use the BulkSplineEvaluator to update
73 /// and read values from the splines in a performant manner.
74 class CompactSpline {
75  public:
76  CompactSpline();
77  CompactSpline(const Range& y_range, const float x_granularity,
78  const int num_nodes = 0);
79 
80  /// The copy constructors, copy operator, and destructor are all default
81  /// implementations, but must be specified explicitly because
82  /// CompactSplineNode is not specified in the header file.
83  CompactSpline(const CompactSpline& rhs);
84  ~CompactSpline();
85  CompactSpline& operator=(const CompactSpline& rhs);
86 
87  /// The range of values for x and y must be specified at spline creation time
88  /// and cannot be changed afterwards. Empties all nodes, if we have any.
89  ///
90  /// @param y_range The upper and lower bounds for y-values in the nodes.
91  /// The more narrow this is, the better the precision of the
92  /// fixed point numbers. Note that you should add 10% padding
93  /// here, since AddNode may insert a smoothing node that is
94  /// slightly beyond the source y range.
95  /// @param x_granularity The minimum increment of x-values. If you're working
96  /// with a spline changes at most 30 times per second,
97  /// and your x is in units of 1/1000th of a second, then
98  /// x_granularity = 33 is a good baseline. You'll
99  /// probably want granularity around 1/50th of that
100  /// baseline value, though, since AddNode may insert
101  /// smoothing nodes at intermediate x's.
102  /// In our example here, you could set
103  /// x_granularity near 33 / 50. For ease of debugging,
104  /// an x_granularity of 0.5 or 1 is probably best.
105  /// @param num_nodes A hint for how many nodes are expected to be added.
106  /// Try to get this close or slightly too high, to avoid
107  /// reallocation of internal memory.
108  void Init(const Range& y_range, const float x_granularity,
109  const int num_nodes = 0);
110 
111  /// Add a node to the end of the spline. Depending on the method, an
112  /// intermediate node may also be inserted.
113  ///
114  /// @param x Must be greater than the x-value of the last spline node. If not,
115  /// this call is a nop.
116  /// @param y Must be within the `y_range` specified in Init().
117  /// @param derivative No restrictions, but excessively large values may still
118  /// result in overshoot, even with an intermediate node.
119  /// @param method If kAddWithoutModification, adds the node and does nothing
120  /// else. If kEnsureCubicWellBehaved, adds the node and
121  /// (if required) inserts another node in the middle so that
122  /// the individual cubics have uniform curvature.
123  /// Uniform curvature means always curving upward or always
124  /// curving downward. See docs/dual_cubics.pdf for details.
125  void AddNode(const float x, const float y, const float derivative,
126  const CompactSplineAddMethod method = kEnsureCubicWellBehaved);
127 
128  /// Add values without converting them. Useful when initializing from
129  /// precalculated data.
130  void AddNodeVerbatim(const CompactSplineXGrain x, const CompactSplineYRung y,
131  const CompactSplineAngle angle);
132 
133  /// Remove all nodes from the spline.
134  void Clear();
135 
136  /// Return index of the first node before `x`.
137  /// If `x` is before the first node, return kBeforeSplineIndex.
138  /// If `x` is past the last node, return kAfterSplineIndex.
139  ///
140  /// @param x x-value in the spline. Most often, the x-axis represents time.
141  /// @param guess_index Best guess at what the index for `x` will be.
142  /// Often the caller will be traversing from low to high x,
143  /// so a good guess is the index after the current index.
144  /// If you have no idea, set to 0.
145  CompactSplineIndex IndexForX(const float x,
146  const CompactSplineIndex guess_index) const;
147 
148  /// If `repeat` is true, loop to x = 0 when `x` >= EndX().
149  /// If `repeat` is false, same as IndexForX().
150  CompactSplineIndex IndexForXAllowingRepeat(
151  const float x, const CompactSplineIndex guess_index,
152  const bool repeat, float* final_x) const;
153 
154  /// Returns closest index between 0 and NumNodes() - 1.
155  /// Clamps `x` to a value in the range of index.
156  /// `index` must be a valid value: i.e. kBeforeSplineIndex, kAfterSplineIndex,
157  /// or between 0..NumNodes()-1.
158  CompactSplineIndex ClampIndex(const CompactSplineIndex index, float* x) const;
159 
160  // First and last x, y, and derivatives in the spline.
161  float StartX() const;
162  float StartY() const;
163  float StartDerivative() const;
164  float EndX() const;
165  float EndY() const;
166  float EndDerivative() const;
167  float NodeX(const CompactSplineIndex index) const;
168  float NodeY(const CompactSplineIndex index) const;
169  float NodeDerivative(const CompactSplineIndex index) const;
170  float LengthX() const { return EndX() - StartX(); }
171  Range RangeX() const { return Range(StartX(), EndX()); }
172  const Range& RangeY() const { return y_range_; }
173 
174  /// Evaluate spline at `x`. This function is somewhat slow because it
175  /// must find the node for `x` and create the cubic before the returned
176  /// y can be evaluated.
177  /// If calling from inside a loop, replace the loop with one call to Ys(),
178  /// which is significantly faster.
179  float YCalculatedSlowly(const float x) const;
180 
181  /// Fast evaluation of a subset of the x-domain of the spline.
182  /// Spline is evaluated from `start_x` and subsequent intervals of `delta_x`.
183  /// Evaluated values are returned in `ys`.
184  void Ys(const float start_x, const float delta_x, const int num_ys,
185  float* ys) const;
186 
187  /// The start and end x-values covered by the segment after `index`.
188  Range RangeX(const CompactSplineIndex index) const;
189 
190  /// Initialization parameters for a cubic curve that starts at `index` and
191  /// ends at `index` + 1. Or a constant curve if `index` is kBeforeSplineIndex
192  /// or kAfterSplineIndex.
193  CubicInit CreateCubicInit(const CompactSplineIndex index) const;
194 
195  /// Returns the number of nodes in this spline.
196  CompactSplineIndex NumNodes() const;
197 
198  /// Return const versions of internal values. For serialization.
199  const CompactSplineNode* nodes() const;
200  const Range& y_range() const { return y_range_; }
201  float x_granularity() const { return x_granularity_; }
202 
203  /// Recommend a granularity given a maximal-x value. We want to have the
204  /// most precise granularity when quantizing x's.
205  static float RecommendXGranularity(const float max_x);
206 
207  /// Fast evaluation of several splines.
208  /// @param splines input splines of length `num_splines`.
209  /// @param num_splines number of splines to evaluate.
210  /// @param start_x starting point for every spline.
211  /// @param delta_x increment for each output y.
212  /// @param num_ys length of the `ys` array.
213  /// @param ys two dimensional output array, ys[num_ys][num_splines].
214  /// ys[0] are `splines` evaluated at start_x.
215  /// ys[num_ys - 1] are `splines` evaluated at
216  /// start_x + delta_x * num_ys.
217  static void BulkYs(const CompactSpline* const splines, const int num_splines,
218  const float start_x, const float delta_x,
219  const size_t num_ys, float* ys);
220 
221  /// Fast evaluation of several splines, with mathfu::VectorPacked interface.
222  /// Useful for evaluate three splines which together form a mathfu::vec3,
223  /// for instance.
224  template <int kDimensions>
225  static void BulkYs(const CompactSpline* const splines, const float start_x,
226  const float delta_x, const size_t num_ys,
227  mathfu::VectorPacked<float, kDimensions>* ys) {
228  BulkYs(splines, kDimensions, start_x, delta_x, num_ys,
229  reinterpret_cast<float*>(ys));
230  }

232  private:
233  CompactSplineIndex LastNodeIndex() const;
234 
235  /// Return true iff `x` is between the the nodes at `index` and `index` + 1.
236  bool IndexContainsX(const CompactSplineXGrain compact_x,
237  const CompactSplineIndex index) const;
238 
239  /// Search the nodes to find the index of the first node before `x`.
240  CompactSplineIndex BinarySearchIndexForX(
241  const CompactSplineXGrain compact_x) const;
242 
243  /// Return e.x - s.x, converted from quantized to external units.
244  float WidthX(const CompactSplineNode& s, const CompactSplineNode& e) const;
245 
246  /// Create the initialization parameters for a cubic running from `s` to `e`.
247  CubicInit CreateCubicInit(const CompactSplineNode& s,
248  const CompactSplineNode& e) const;
249 
250  /// Array of key points (x, y, derivative) that describe the curve.
251  /// The curve is interpolated smoothly between these key points.
252  /// Key points are stored in quantized form, and converted back to world
253  /// co-ordinates by using `y_range_` and `x_granularity_`.
254  std::vector<CompactSplineNode> nodes_;
255 
256  /// Extreme values for y. See comments on Init() for details.
257  Range y_range_;
258 
259  /// Minimum increment for x. See comments on Init() for details.
260  float x_granularity_;
261 };

263 /// @class SplinePlayback
264 /// @brief Parameters to specify how a spline should be traversed.
265 struct SplinePlayback {
266  /// Initialize all channels with same spline.
267  /// Especially useful when kDimensions = 1, since there is only one channel.
268  explicit SplinePlayback(float start_x = 0.0f, bool repeat = false,
269  float playback_rate = 1.0f, float blend_x = 0.0f)
270  : start_x(start_x), blend_x(blend_x), playback_rate(playback_rate),
271  repeat(repeat) {}
272 
273  /// The starting point from which to play.
274  float start_x;
275 
276  /// The point at which to be 100% in this spline. We create a smooth spline
277  /// from the current state to the spline state that lasts for `blend_x`.
278  float blend_x;
279 
280  /// The playback rate of the spline. Scales `delta_time` of the update to
281  /// to x-axis of `splines`.
282  /// 0 ==> paused
283  /// 0.5 ==> half speed (slow motion)
284  /// 1 ==> authored speed
285  /// 2 ==> double speed (fast forward)
286  float playback_rate;
287 
288  /// If true, start back at the beginning after we reach the end.
289  bool repeat;
290 };
