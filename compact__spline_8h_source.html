62 /// @class CompactSpline
63 /// @brief Represent a smooth curve in a small amount of memory.
64 ///
65 /// This spline interpolates a series of (x, y, derivative) nodes to create a
66 /// smooth curve.
67 ///
68 /// This class holds a series of such nodes, and aids with the construction of
69 /// that series by inserting extra nodes when extra smoothness is required.
70 ///
71 /// The data in this class is compacted as quantized values. It's not intended
72 /// to be read directly. You should use the BulkSplineEvaluator to update
73 /// and read values from the splines in a performant manner.
74 class CompactSpline {
75  public:
76  CompactSpline();
77  CompactSpline(const Range& y_range, const float x_granularity,
78  const int num_nodes = 0);
79 
80  /// The copy constructors, copy operator, and destructor are all default
81  /// implementations, but must be specified explicitly because
82  /// CompactSplineNode is not specified in the header file.
83  CompactSpline(const CompactSpline& rhs);
84  ~CompactSpline();
85  CompactSpline& operator=(const CompactSpline& rhs);
86 
87  /// The range of values for x and y must be specified at spline creation time
88  /// and cannot be changed afterwards. Empties all nodes, if we have any.
89  ///
90  /// @param y_range The upper and lower bounds for y-values in the nodes.
91  /// The more narrow this is, the better the precision of the
92  /// fixed point numbers. Note that you should add 10% padding
93  /// here, since AddNode may insert a smoothing node that is
94  /// slightly beyond the source y range.
95  /// @param x_granularity The minimum increment of x-values. If you're working
96  /// with a spline changes at most 30 times per second,
97  /// and your x is in units of 1/1000th of a second, then
98  /// x_granularity = 33 is a good baseline. You'll
99  /// probably want granularity around 1/50th of that
100  /// baseline value, though, since AddNode may insert
101  /// smoothing nodes at intermediate x's.
102  /// In our example here, you could set
103  /// x_granularity near 33 / 50. For ease of debugging,
104  /// an x_granularity of 0.5 or 1 is probably best.
105  /// @param num_nodes A hint for how many nodes are expected to be added.
106  /// Try to get this close or slightly too high, to avoid
107  /// reallocation of internal memory.
108  void Init(const Range& y_range, const float x_granularity,
109  const int num_nodes = 0);
110 
111  /// Add a node to the end of the spline. Depending on the method, an
112  /// intermediate node may also be inserted.
113  ///
114  /// @param x Must be greater than the x-value of the last spline node. If not,
115  /// this call is a nop.
116  /// @param y Must be within the `y_range` specified in Init().
117  /// @param derivative No restrictions, but excessively large values may still
118  /// result in overshoot, even with an intermediate node.
119  void AddNode(const float x, const float y, const float derivative,
120  const CompactSplineAddMethod method = kEnsureCubicWellBehaved);
121 
122  /// Remove all nodes from the spline.
123  void Clear();
124 
125  /// Return index of the first node before `x`.
126  /// If `x` is before the first node, return kBeforeSplineIndex.
127  /// If `x` is past the last node, return kAfterSplineIndex.
128  ///
129  /// @param x x-value in the spline. Most often, the x-axis represents time.
130  /// @param guess_index Best guess at what the index for `x` will be.
131  /// Often the caller will be traversing from low to high x,
132  /// so a good guess is the index after the current index.
133  /// If you have no idea, set to 0.
134  CompactSplineIndex IndexForX(const float x,
135  const CompactSplineIndex guess_index) const;
136 
137  // First and last x, y, and derivatives in the spline.
138  float StartX() const;
139  float StartY() const;
140  float StartDerivative() const;
141  float EndX() const;
142  float EndY() const;
143  float EndDerivative() const;
144  float NodeX(const CompactSplineIndex index) const;
145  float NodeY(const CompactSplineIndex index) const;
146  float NodeDerivative(const CompactSplineIndex index) const;
147  float LengthX() const { return EndX() - StartX(); }
148  Range RangeX() const { return Range(StartX(), EndX()); }
149  const Range& RangeY() const { return y_range_; }
150 
151  /// The start and end x-values covered by the segment after `index`.
152  Range RangeX(const CompactSplineIndex index) const;
153 
154  /// Initialization parameters for a cubic curve that starts at `index` and
155  /// ends at `index` + 1. Or a constant curve if `index` is kBeforeSplineIndex
156  /// or kAfterSplineIndex.
157  CubicInit CreateCubicInit(const CompactSplineIndex index) const;
158 
159  /// Recommend a granularity given a maximal-x value. We want to have the
160  /// most precise granularity when quantizing x's.
161  static float RecommendXGranularity(const float max_x);
162 
163  private:
164  CompactSplineIndex LastNodeIndex() const;
165 
166  /// Return true iff `x` is between the the nodes at `index` and `index` + 1.
167  bool IndexContainsX(const CompactSplineXGrain compact_x,
168  const CompactSplineIndex index) const;
169 
170  /// Search the nodes to find the index of the first node before `x`.
171  CompactSplineIndex BinarySearchIndexForX(
172  const CompactSplineXGrain compact_x) const;
173 
174  /// Return e.x - s.x, converted from quantized to external units.
175  float WidthX(const CompactSplineNode& s, const CompactSplineNode& e) const;
176 
177  /// Create the initialization parameters for a cubic running from `s` to `e`.
178  CubicInit CreateCubicInit(const CompactSplineNode& s,
179  const CompactSplineNode& e) const;
180 
181  /// Array of key points (x, y, derivative) that describe the curve.
182  /// The curve is interpolated smoothly between these key points.
183  /// Key points are stored in quantized form, and converted back to world
184  /// co-ordinates by using `y_range_` and `x_granularity_`.
185  std::vector<CompactSplineNode> nodes_;
186 
187  /// Extreme values for y. See comments on Init() for details.
188  Range y_range_;
189 
190  /// Minimum increment for x. See comments on Init() for details.
191  float x_granularity_;
192 };
