57 // A spline interpolates a series of (x, y, derivative) nodes to create a
58 // smooth curve.
59 //
60 // This class holds a series of such nodes, and aids with the construction of
61 // that series by inserting extra nodes when extra smoothness is required.
62 //
63 // The data in this class is compacted as quantized values. It's not intended
64 // to be read directly. You should use the BulkSplineEvaluator to update
65 // and read values from the splines in a performant manner.
66 class CompactSpline {
67  public:
68  CompactSpline();
69  CompactSpline(const Range& y_range, const float x_granularity,
70  const int num_nodes = 0);
71 
72  // The copy constructors, copy operator, and destructor are all default
73  // implementations, but must be specified explicitly because
74  // CompactSplineNode is not specified in the header file.
75  CompactSpline(const CompactSpline& rhs);
76  ~CompactSpline();
77  CompactSpline& operator=(const CompactSpline& rhs);
78 
79  // The range of values for x and y must be specified at spline creation time
80  // and cannot be changed afterwards. Empties all nodes, if we have any.
81  //
82  // 'y_range' -- The upper and lower bounds for y-values in the nodes.
83  // The more narrow this is, the better the precision of the
84  // fixed point numbers. Note that you should add 10% padding
85  // here, since AddNode may insert a smoothing node that is
86  // slightly beyond the source y range.
87  // 'x_granularity' -- The minimum increment of x-values. If you're working
88  // with a spline changes at most 30 times per second,
89  // and your x is in units of 1/1000th of a second, then
90  // x_granularity = 33 is a good baseline. You'll probably
91  // want granularity around 1/50th of that baseline value,
92  // though, since AddNode may insert smoothing nodes at
93  // intermediate x's. In our example here, you could set
94  // x_granularity near 33 / 50. For ease of debugging,
95  // an x_granularity of 0.5 or 1 is probably best.
96  // 'num_nodes' -- A hint for how many nodes are expected to be added.
97  // Try to get this close or slightly too high, to avoid
98  // reallocation of internal memory.
99  void Init(const Range& y_range, const float x_granularity,
100  const int num_nodes = 0);
101 
102  // Add a node to the end of the spline. Depending on the method, an
103  // intermediate node may also be inserted.
104  //
105  // 'x' -- Must be greater than the x-value of the last spline node. If not,
106  // this call is a nop.
107  // 'y' -- Must be within the 'y_range' specified in Init().
108  // 'derivative' -- No restrictions, but excessively large values may still
109  // result in overshoot, even with an intermediate node.
110  void AddNode(const float x, const float y, const float derivative,
111  const CompactSplineAddMethod method = kEnsureCubicWellBehaved);
112 
113  // Remove all nodes from the spline.
114  void Clear();
115 
116  // Return index of the first node before 'x'.
117  // If 'x' is before the first node, return kBeforeSplineIndex.
118  // If 'x' is past the last node, return kAfterSplineIndex.
119  //
120  // 'guess_index' -- Best guess at what the index for 'x' will be. Often the
121  // caller will be traversing from low to high x, so a good
122  // guess is the index after the current index.
123  // If you have no idea, set to 0.
124  CompactSplineIndex IndexForX(const float x,
125  const CompactSplineIndex guess_index) const;
126 
127  // First and last x, y, and derivatives in the spline.
128  float StartX() const;
129  float StartY() const;
130  float StartDerivative() const;
131  float EndX() const;
132  float EndY() const;
133  float EndDerivative() const;
134  float NodeX(const CompactSplineIndex index) const;
135  float NodeY(const CompactSplineIndex index) const;
136  float NodeDerivative(const CompactSplineIndex index) const;
137  float LengthX() const { return EndX() - StartX(); }
138  Range RangeX() const { return Range(StartX(), EndX()); }
139  const Range& RangeY() const { return y_range_; }
140 
141  // The start and end x-values covered by the segment after 'index'.
142  Range RangeX(const CompactSplineIndex index) const;
143 
144  // Initialization parameters for a cubic curve that starts at 'index' and
145  // ends at 'index' + 1. Or a constant curve if 'index' is kBeforeSplineIndex
146  // or kAfterSplineIndex.
147  CubicInit CreateCubicInit(const CompactSplineIndex index) const;
148 
149  static float RecommendXGranularity(const float max_x);
150 
151  private:
152  CompactSplineIndex LastNodeIndex() const;
153 
154  // Return true iff 'x' is between the the nodes at 'index' and 'index' + 1.
155  bool IndexContainsX(const CompactSplineXGrain compact_x,
156  const CompactSplineIndex index) const;
157 
158  // Search the nodes to find the index of the first node before 'x'.
159  CompactSplineIndex BinarySearchIndexForX(
160  const CompactSplineXGrain compact_x) const;
161 
162  // Return e.x - s.x, converted from quantized to external units.
163  float WidthX(const CompactSplineNode& s, const CompactSplineNode& e) const;
164 
165  // Create the initialization parameters for a cubic running from 's' to 'e'.
166  CubicInit CreateCubicInit(const CompactSplineNode& s,
167  const CompactSplineNode& e) const;
168 
169  // Array of key points (x, y, derivative) that describe the curve.
170  // The curve is interpolated smoothly between these key points.
171  // Key points are stored in quantized form, and converted back to world
172  // co-ordinates by using 'y_range_' and 'x_granularity_'.
173  std::vector<CompactSplineNode> nodes_;
174 
175  // Extreme values for y. See comments on Init() for details.
176  Range y_range_;
177 
178  // Minimum increment for x. See comments on Init() for details.
179  float x_granularity_;
180 };
