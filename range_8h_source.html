23 // If using modular arithmetic, there are two paths to the target: one that
24 // goes directly and one that wraps around. This enum represents differet ways
25 // to choose the path.
26 enum ModularDirection {
27  kDirectionClosest,
28  kDirectionFarthest,
29  kDirectionPositive,
30  kDirectionNegative,
31  kDirectionDirect,

34 /// Represent an interval on a number line.
35 template <class T>
36 class RangeT {
37  public:
38  template <size_t kMaxLen>
39  struct TArray {
40  size_t len;

50  // By default, initialize to an invalid range.
51  RangeT() : start_(static_cast<T>(1)), end_(static_cast<T>(0)) {}
52  RangeT(const T start, const T end) : start_(start), end_(end) {}
53 
54  // A range is valid if it contains at least one number.
55  bool Valid() const { return start_ <= end_; }
56 
57  // Returns the mid-point of the range, rounded down for integers.
58  // Behavior is undefined for invalid regions.
59  T Middle() const { return (start_ + end_) / static_cast<T>(2); }
60 
61  // Returns the span of the range. Returns 0 when only one number in range.
62  // Behavior is undefined for invalid regions.
63  T Length() const { return end_ - start_; }
64 
65  // Returns 'x' if it is within the range. Otherwise, returns start_ or end_,
66  // whichever is closer to 'x'.
67  // Behavior is undefined for invalid regions.
68  T Clamp(const T x) const { return mathfu::Clamp(x, start_, end_); }
69 
70  // Clamp 'x' so it is inside one of the bounds. Can save cycles if you
71  // already know that 'x' is inside the other bound.
72  T ClampAfterStart(const T x) const { return std::max(x, start_); }
73  T ClampBeforeEnd(const T x) const { return std::min(x, end_); }
74 
75  // Returns distances outside of the range. If inside the range, returns 0.
76  // Behavior is undefined for invalid regions.
77  T DistanceFrom(const T x) const { return fabs(x - Clamp(x)); }
78 
79  // Lerps between the start and the end.
80  // 'percent' of 0 returns start. 'percent' of 1 returns end.
81  // Behavior is undefined for invalid regions.
82  T Lerp(const float percent) const {
83  return mathfu::Lerp(start_, end_, percent);
84  }
85 
86  // Returns percent 0~1, from start to end. *Not* clamped to 0~1.
87  // 0 ==> start; 1 ==> end; 0.5 ==> Middle(); -1 ==> start - Length()
88  float Percent(const T x) const { return (x - start_) / Length(); }
89 
90  // Returns percent 0~1, from start to end. Clamped to 0~1.
91  // 0 ==> start or earlier; 1 ==> end or later; 0.5 ==> Middle()
92  T PercentClamped(const T x) const {
93  return mathfu::Clamp(Percent(x), 0.0f, 1.0f);
94  }

96  // Ensure 'x' is within the valid constraint range.
97  // 'x' must be within +-Length() of the range bounds. This is a reasonable
98  // restriction in most cases (such as after an arithmetic operation).
99  // For cases where 'x' may be wildly outside the range, use
100  // NormalizeWildValue() instead.
101  T Normalize(T x) const { return x + ModularAdjustment(x); }
102 
103  T NormalizeWildValue(T x) const {
104  // Use (expensive) division to determine how many lengths we are away from
105  // the normalized range.
106  const T length = Length();
107  const T units = (x - start_) / length;
108  const T whole_units = floor(units);
109 
110  // Subtract off those units to get something that (mathematically) should
111  // be normalized. Due to Ting point error, it sometimes is slightly
112  // outside the bounds, so we need to do a standard normalization afterwards.
113  const T close = x - whole_units * length;
114  const T normalized = close + ModularAdjustment(close);
115  return normalized;
116  }
117 
118  // Returns:
119  // Length() if 'x' is below the valid range
120  // -Length() if 'x' is above the valid range
121  // 0 if 'x' is within the valid range.
122  T ModularAdjustment(T x) const {
123  const T length = Length();
124  const T adjustment = x <= start_ ? length : x > end_ ? -length : 0.0f;
125  assert(start_ < x + adjustment && x + adjustment <= end_);
126  return adjustment;
127  }
128 
129  // In modular arithmetic, you can get from 'a' to 'b' by going directly, or
130  // by wrapping around.
131  // Return the closest difference from 'a' to 'b' under modular arithmetic.
132  float ModDiffClose(T a, T b) const { return Normalize(b - a); }
133 
134  // Return the farthest difference from 'a' to 'b' under modular arithmetic.
135  float ModDiffFar(T a, T b) const {
136  const float length = Length();
137  const float close = ModDiffClose(a, b);
138  return close >= 0.0f ? close - length : close + length;
139  }
140 
141  // Return the difference from 'a' to 'b' under modular arithmetic that is
142  // positive.
143  float ModDiffPositive(T a, T b) const {
144  const float length = Length();
145  const float close = ModDiffClose(a, b);
146  return close >= 0.0f ? close : close + length;
147  }
148 
149  // Return the difference from 'a' to 'b' under modular arithmetic that is
150  // negative.
151  float ModDiffNegative(T a, T b) const {
152  const float length = Length();
153  const float close = ModDiffClose(a, b);
154  return close >= 0.0f ? close - length : close;
155  }
156 
157  // Return the difference from 'a' to 'b' that satisfies the 'direction'
158  // criteria.
159  float ModDiff(T a, T b, ModularDirection direction) const {
160  switch (direction) {
161  case kDirectionClosest:
162  return ModDiffClose(a, b);
163  case kDirectionFarthest:
164  return ModDiffFar(a, b);
165  case kDirectionPositive:
166  return ModDiffPositive(a, b);
167  case kDirectionNegative:
168  return ModDiffNegative(a, b);
169  case kDirectionDirect:
170  return b - a;
171  }
172  assert(false);
173  return 0.0f;
174  }

176  bool Contains(const T x) const { return start_ <= x && x <= end_; }
177 
178  // Swap start and end. When 'a' and 'b' don't overlap, if you invert the
179  // return value of Range::Intersect(a, b), you'll get the gap between
180  // 'a' and 'b'.
181  RangeT Invert() const { return RangeT(end_, start_); }
182 
183  // Returns a range that is 'percent' longer. If 'percent' is < 1.0, then
184  // returned range will actually be shorter.
185  RangeT Lengthen(const float percent) const {
186  const T extra = static_cast<T>(Length() * percent * 0.5f);
187  return RangeT(start_ - extra, end_ + extra);
188  }

205  // Return the overlap of 'a' and 'b', or an invalid range if they do not
206  // overlap at all.
207  // When 'a' and 'b' don't overlap at all, calling Invert on the returned
208  // range will give the gap between 'a' and 'b'.
209  static RangeT Intersect(const RangeT& a, const RangeT& b) {
210  // Possible cases:
211  // 1. |-a---| |-b---| ==> return invalid
212  // 2. |-b---| |-a---| ==> return invalid
213  // 3. |-a---------| ==> return b
214  // |-b---|
215  // 4. |-b---------| ==> return a
216  // |-a---|
217  // 5. |-a---| ==> return (b.start, a.end)
218  // |-b---|
219  // 6. |-b---| ==> return (a.start, b.end)
220  // |-a---|
221  //
222  // All satisfied by,
223  // intersection.start = max(a.start, b.start)
224  // intersection.end = min(a.end, b.end)
225  // Note that ranges where start > end are considered invalid.
226  return RangeT(std::max(a.start_, b.start_), std::min(a.end_, b.end_));
227  }

229  // Only keep entries in 'values' if they are in
230  // (range.start - epsition, range.end + epsilon).
231  // Any values that are kept are clamped to 'range'.
232  //
233  // This function is useful when floating point precision error might put a
234  // value slightly outside 'range' even though mathematically it should be
235  // inside 'range'. This often happens with values right on the border of the
236  // valid range.
237  static size_t ValuesInRange(const RangeT& range, T epsilon, size_t num_values,
238  T* values) {
239  size_t num_returned = 0;
240  for (size_t i = 0; i < num_values; ++i) {
241  const T value = values[i];
242  const T clamped = range.Clamp(value);
243  const T dist = fabs(value - clamped);
244 
245  // If the distance from the range is small, keep the clamped value.
246  if (dist <= epsilon) {
247  values[num_returned++] = clamped;
248  }
249  }
250  return num_returned;
251  }

259  // Intersect every element of 'a' with every element of 'b'. Append
260  // intersections to 'intersections'. Note that 'intersections' is not reset at
261  // the start of the call.
262  static size_t IntersectRanges(const RangeT* a, size_t len_a, const RangeT* b,
263  size_t len_b, RangeT* intersections,
264  RangeT* gaps = nullptr,
265  size_t* len_gaps = nullptr) {
266  size_t num_intersections = 0;
267  size_t num_gaps = 0;
268 
269  for (size_t i = 0; i < len_a; ++i) {
270  for (size_t j = 0; j < len_b; ++j) {
271  const RangeT intersection = RangeT::Intersect(a[i], b[j]);
272  if (intersection.Valid()) {
273  intersections[num_intersections++] = intersection;
274 
275  } else if (gaps != nullptr) {
276  // Return the gaps, too, if requested. Invert() invalid intersections
277  // to get the gap between the ranges.
278  gaps[num_gaps++] = intersection.Invert();
279  }
280  }
281  }
282 
283  // Set return values.
284  if (len_gaps != nullptr) {
285  *len_gaps = num_gaps;
286  }
287  return num_intersections;
288  }

290  template <size_t kMaxLen>
291  static void IntersectRanges(const RangeArray<kMaxLen>& a,
292  const RangeArray<kMaxLen>& b,
293  RangeArray<kMaxLen * kMaxLen>* intersections,
294  RangeArray<kMaxLen * kMaxLen>* gaps = nullptr) {
295  const bool use_gaps = gaps != nullptr;
296  intersections->len = IntersectRanges(
297  a.arr, a.len, b.arr, b.len, intersections->arr,
298  use_gaps ? gaps->arr : nullptr, use_gaps ? &gaps->len : nullptr);
299  }

301  // Return the index of the longest range in 'ranges'.
302  static size_t IndexOfLongest(const RangeT* ranges, size_t len) {
303  T longest_length = -1.0f;
304  size_t longest_index = 0;
305  for (size_t i = 0; i < len; ++i) {
306  const T length = ranges[i].Length();
307  if (length > longest_length) {
308  longest_length = length;
309  longest_index = i;
310  }
311  }
312  return longest_index;
313  }

317  return IndexOfLongest(ranges.arr, ranges.len);
