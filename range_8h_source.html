51  // By default, initialize to an invalid range.
52  RangeT() : start_(static_cast<T>(1)), end_(static_cast<T>(0)) {}
53  RangeT(const T start, const T end) : start_(start), end_(end) {}

55  /// A range is valid if it contains at least one number.
56  bool Valid() const { return start_ <= end_; }
57 
58  /// Returns the mid-point of the range, rounded down for integers.
59  /// Behavior is undefined for invalid regions.
60  T Middle() const { return (start_ + end_) / static_cast<T>(2); }
61 
62  /// Returns the span of the range. Returns 0 when only one number in range.
63  /// Behavior is undefined for invalid regions.
64  T Length() const { return end_ - start_; }
65 
66  /// Returns `x` if it is within the range. Otherwise, returns start_ or end_,
67  /// whichever is closer to `x`.
68  /// Behavior is undefined for invalid regions.
69  T Clamp(const T x) const { return mathfu::Clamp(x, start_, end_); }
70 
71  /// Clamp `x` so it is inside the start bound. Can save cycles if you
72  /// already know that `x` is inside the end bound.
73  T ClampAfterStart(const T x) const { return std::max(x, start_); }
74 
75  /// Clamp `x` so it is inside the end bound. Can save cycles if you
76  /// already know that `x` is inside the start bound.
77  T ClampBeforeEnd(const T x) const { return std::min(x, end_); }
78 
79  /// Returns distance outside of the range. If inside the range, returns 0.
80  /// Behavior is undefined for invalid regions.
81  T DistanceFrom(const T x) const { return fabs(x - Clamp(x)); }
82 
83  /// Lerps between the start and the end.
84  /// 'percent' of 0 returns start. 'percent' of 1 returns end.
85  /// Behavior is undefined for invalid regions.
86  T Lerp(const float percent) const {
87  return mathfu::Lerp(start_, end_, percent);
88  }

90  /// Returns percent 0~1, from start to end. *Not* clamped to 0~1.
91  /// 0 ==> start; 1 ==> end; 0.5 ==> Middle(); -1 ==> start - Length()
92  float Percent(const T x) const { return (x - start_) / Length(); }
93 
94  /// Returns percent 0~1, from start to end. Clamped to 0~1.
95  /// 0 ==> start or earlier; 1 ==> end or later; 0.5 ==> Middle()
96  T PercentClamped(const T x) const {
97  return mathfu::Clamp(Percent(x), 0.0f, 1.0f);

100  /// Ensure `x` is within the valid constraint range, by subtracting or
101  /// adding Length() to it.
102  /// `x` must be within +-Length() of the range bounds. This is a reasonable
103  /// restriction in most cases (such as after an arithmetic operation).
104  /// For cases where `x` may be wildly outside the range, use
105  /// NormalizeCloseValue() or NormalizeWildValue() instead.
106  T Normalize(T x) const {
107  const float normalized = x + ModularAdjustment(x);
108  assert(ContainsExcludingStart(normalized));
109  return normalized;
110  }
111 
112  /// Ensure `x` is within the valid constraint range, by subtracting or
113  /// adding Length() to it repeatedly.
114  /// `x` must be within +-kMaxAdjustments * Length() of the range bounds
115  /// for this function to be effective. If its greater, we end up calling
116  /// NormalizeWildValue(), so you'd be better off calling NormalizeWildValue()
117  /// from the beginning.
118  /// This function is intended to be called in situations where `x` is almost
119  /// always within one or two lengths of being normalized, so we don't want to
120  /// incur the division cost of NormalizeWildValue(). It's still guaranteed
121  /// to return a normalized value, however.
122  T NormalizeCloseValue(T x) const {
123  static const int kMaxAdjustments = 4;
124 
125  // Return without change if `x` is already normalized.
126  const bool below = x <= start_;
127  const bool above = x > end_;
128  if (!below && !above) return x;
129 
130  // Each time through the loop, we'll adjust by one length closer to the
131  // valid interval.
132  const T length = Length();
133  int num_adjustments = 0;
134 
135  if (below) {
136  // Keep adding until we're in the range.
137  do {
138  x += length;
139  if (num_adjustments++ > kMaxAdjustments) return NormalizeWildValue(x);
140  } while (x <= start_);
141  } else {
142  // Keep subtracting until we're in the range.
143  do {
144  x -= length;
145  if (num_adjustments++ > kMaxAdjustments) return NormalizeWildValue(x);
146  } while (x > end_);
147  }
148  return x;
149  }
150 
151  /// Ensure `x` is within the valid constraint range, by subtracting multiples
152  /// of Length() from it until it is.
153  /// `x` can be any value.
154  T NormalizeWildValue(T x) const {
155  // Use (expensive) division to determine how many lengths we are away from
156  // the normalized range.
157  const T length = Length();
158  const T units = (x - start_) / length;
159  const T whole_units = floor(units);
160 
161  // Subtract off those units to get something that (mathematically) should
162  // be normalized. Due to Ting point error, it sometimes is slightly
163  // outside the bounds, so we need to do a standard normalization afterwards.
164  const T close = x - whole_units * length;
165  const T normalized = close + ModularAdjustment(close);
166  assert(ContainsExcludingStart(normalized));
167  return normalized;
168  }
169 
170  /// Returns:
171  /// Length() if `x` is below the valid range
172  /// -Length() if `x` is above the valid range
173  /// 0 if `x` is within the valid range.
174  T ModularAdjustment(T x) const {
175  const T length = Length();
176  const T adjustment = x <= start_ ? length : x > end_ ? -length : 0.0f;
177  return adjustment;
178  }

180  /// In modular arithmetic, you can get from 'a' to 'b' by going directly, or
181  /// by wrapping around.
182  /// Return the closest difference from 'a' to 'b' under modular arithmetic.
183  float ModDiffClose(T a, T b) const { return Normalize(b - a); }
184 
185  /// Return the farthest difference from 'a' to 'b' under modular arithmetic.
186  float ModDiffFar(T a, T b) const {
187  const float length = Length();
188  const float close = ModDiffClose(a, b);
189  return close >= 0.0f ? close - length : close + length;
190  }
191 
192  /// Return the difference from 'a' to 'b' under modular arithmetic that is
193  /// positive.
194  float ModDiffPositive(T a, T b) const {
195  const float length = Length();
196  const float close = ModDiffClose(a, b);
197  return close >= 0.0f ? close : close + length;
198  }
199 
200  /// Return the difference from 'a' to 'b' under modular arithmetic that is
201  /// negative.
202  float ModDiffNegative(T a, T b) const {
203  const float length = Length();
204  const float close = ModDiffClose(a, b);
205  return close >= 0.0f ? close - length : close;
206  }

208  /// Return the difference from 'a' to 'b' that satisfies the 'direction'
209  /// criteria.
210  float ModDiff(T a, T b, ModularDirection direction) const {
211  switch (direction) {
212  case kDirectionClosest:
213  return ModDiffClose(a, b);
214  case kDirectionFarthest:
215  return ModDiffFar(a, b);
216  case kDirectionPositive:
217  return ModDiffPositive(a, b);
218  case kDirectionNegative:
219  return ModDiffNegative(a, b);
220  case kDirectionDirect:
221  return b - a;
222  }
223  assert(false);
224  return 0.0f;
225  }

276  /// Return the overlap of 'a' and 'b', or an invalid range if they do not
277  /// overlap at all.
278  /// When 'a' and 'b' don't overlap at all, calling Invert on the returned
279  /// range will give the gap between 'a' and 'b'.
280  static RangeT Intersect(const RangeT& a, const RangeT& b) {
281  // Possible cases:
282  // 1. |-a---| |-b---| ==> return invalid
283  // 2. |-b---| |-a---| ==> return invalid
284  // 3. |-a---------| ==> return b
285  // |-b---|
286  // 4. |-b---------| ==> return a
287  // |-a---|
288  // 5. |-a---| ==> return (b.start, a.end)
289  // |-b---|
290  // 6. |-b---| ==> return (a.start, b.end)
291  // |-a---|
292  //
293  // All satisfied by,
294  // intersection.start = max(a.start, b.start)
295  // intersection.end = min(a.end, b.end)
296  // Note that ranges where start > end are considered invalid.
297  return RangeT(std::max(a.start_, b.start_), std::min(a.end_, b.end_));
298  }

300  /// Return the smallest range that covers all of 'a' and 'b'.
301  static RangeT Union(const RangeT& a, const RangeT& b) {
302  // Possible cases:
303  // 1. |-a---| |-b---| ==> return (a.start, b.end)
304  // 2. |-b---| |-a---| ==> return (b.start, a.end)
305  // 3. |-a---------| ==> return a
306  // |-b---|
307  // 4. |-b---------| ==> return b
308  // |-a---|
309  // 5. |-a---| ==> return (a.start, b.end)
310  // |-b---|
311  // 6. |-b---| ==> return (b.start, a.end)
312  // |-a---|
313  //
314  // All satisfied by,
315  // intersection.start = min(a.start, b.start)
316  // intersection.end = max(a.end, b.end)
317  return RangeT(std::min(a.start_, b.start_), std::max(a.end_, b.end_));
318  }

320  /// Only keep entries in 'values' if they are in
321  /// (range.start - epsition, range.end + epsilon).
322  /// Any values that are kept are clamped to 'range'.
323  ///
324  /// This function is useful when floating point precision error might put a
325  /// value slightly outside 'range' even though mathematically it should be
326  /// inside 'range'. This often happens with values right on the border of the
327  /// valid range.
328  static size_t ValuesInRange(const RangeT& range, T epsilon, size_t num_values,
329  T* values) {
330  size_t num_returned = 0;
331  for (size_t i = 0; i < num_values; ++i) {
332  const T value = values[i];
333  const T clamped = range.Clamp(value);
334  const T dist = fabs(value - clamped);
335 
336  // If the distance from the range is small, keep the clamped value.
337  if (dist <= epsilon) {
338  values[num_returned++] = clamped;
339  }
340  }
341  return num_returned;
342  }

350  /// Intersect every element of 'a' with every element of 'b'. Append
351  /// intersections to 'intersections'. Note that 'intersections' is not reset
352  /// at the start of the call.
353  static size_t IntersectRanges(const RangeT* a, size_t len_a, const RangeT* b,
354  size_t len_b, RangeT* intersections,
355  RangeT* gaps = nullptr,
356  size_t* len_gaps = nullptr) {
357  size_t num_intersections = 0;
358  size_t num_gaps = 0;
359 
360  for (size_t i = 0; i < len_a; ++i) {
361  for (size_t j = 0; j < len_b; ++j) {
362  const RangeT intersection = RangeT::Intersect(a[i], b[j]);
363  if (intersection.Valid()) {
364  intersections[num_intersections++] = intersection;
365 
366  } else if (gaps != nullptr) {
367  // Return the gaps, too, if requested. Invert() invalid intersections
368  // to get the gap between the ranges.
369  gaps[num_gaps++] = intersection.Invert();
370  }
371  }
372  }
373 
374  // Set return values.
375  if (len_gaps != nullptr) {
376  *len_gaps = num_gaps;
377  }
378  return num_intersections;
379  }

382  static void IntersectRanges(const RangeArray<kMaxLen>& a,
383  const RangeArray<kMaxLen>& b,
384  RangeArray<kMaxLen * kMaxLen>* intersections,
385  RangeArray<kMaxLen * kMaxLen>* gaps = nullptr) {
386  const bool use_gaps = gaps != nullptr;

389  use_gaps ? gaps->arr : nullptr, use_gaps ? &gaps->len : nullptr);

395  size_t longest_index = 0;

397  const T length = ranges[i].Length();

430  /// Return the index of the shortest range in `ranges`.
431  static T ClampToClosest(T x, const RangeT* ranges, size_t len) {
432  T closest_dist = std::numeric_limits<T>::infinity();
433  T closest_clamp = x;
434  for (size_t i = 0; i < len; ++i) {
435  const T clamp = ranges[i].Clamp(x);
436  const T dist = fabs(x - clamp);
437  if (dist < closest_dist) {
438  closest_dist = dist;
439  closest_clamp = clamp;
440  }
441  }
442  return closest_clamp;
443  }
