30 /// @class MotiveEngine
31 /// @brief Hold and update all animation data.
32 ///
33 /// The engine holds all of the MotiveProcessors, and updates them all when
34 /// AdvanceFrame() is called. The processing is kept central, in this manner,
35 /// for scalability. The engine is not a singleton, but you should try to
36 /// minimize the number of engines in your game. As more Motivators are added to
37 /// the processors, you start to get economies of scale.
38 class MotiveEngine {
39  struct ProcessorDetails {
40  MotiveProcessor* processor;
41  int benchmark_id;
42  bool operator<(const ProcessorDetails& rhs) const {
43  return processor->Priority() < rhs.processor->Priority();
44  }
45  };
46  typedef std::map<MotivatorType, MotiveProcessor*> ProcessorMap;
47  typedef std::pair<MotivatorType, MotiveProcessor*> ProcessorPair;
48  typedef std::multiset<ProcessorDetails> ProcessorSet;
49  typedef std::map<MotivatorType, MotiveProcessorFunctions> FunctionMap;
50  typedef std::pair<MotivatorType, MotiveProcessorFunctions> FunctionPair;
51 
52  public:
53  MotiveEngine();
54  ~MotiveEngine() { Reset(); }
55 
56  /// Deallocate all MotiveProcessors, which, in turn, resets all Motivators
57  /// that use those MotiveProcessors.
58  void Reset();
59 
60  /// Update all the MotiveProcessors by 'delta_time'. This advances all
61  /// Motivators created with this MotiveEngine.
62  /// @param delta_time Elapsed time since the last call to AdvanceFrame().
63  /// Time units are defined by the user. When using spline
64  /// animations, for instance, the time unit is the unit of
65  /// the x-axis.
66  void AdvanceFrame(MotiveTime delta_time);
67 
68  /// @private For internal use only.
69  MotiveProcessor* Processor(MotivatorType type);
70 
71  /// @private For internal use only.
72  static void RegisterProcessorFactory(MotivatorType type,
73  const MotiveProcessorFunctions& fns);
74 
75  private:
76  /// Map from the MotivatorType to the MotiveProcessor. Only one
77  /// MotiveProcessor per type per engine. This is to maximize centralization
78  /// of data.
79  ProcessorMap mapped_processors_;
80 
81  /// Sort the MotiveProcessors by priority. Low numbered priorities run first.
82  /// This allows high number priorities to have child motivators, as long as
83  /// the child motivators have lower priority.
84  ProcessorSet sorted_processors_;
85 
86  /// Current version of the Motive Animation System.
87  const MotiveVersion* version_;
88 
89  /// ProcessorMap from the MotivatorType to the factory that creates the
90  /// MotiveProcessor. We only create an MotiveProcessor when one is needed.
91  static FunctionMap function_map_;
92 };
