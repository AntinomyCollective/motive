29 /// @class AnimTable
30 /// @brief Hold animation lists for several object types.
31 class AnimTable {
32  public:
33  /// Callback that returns a pointer to the `RigAnimFb` FlatBuffer table for
34  /// `anim_name`. Your implementation may vary, but one possibility is to
35  /// load the file with `anim_name` into `scratch_buf`, and then return a
36  /// pointer to `scratch_buf`.
37  ///
38  /// Note: Motive does not make assumptions on file io, so the caller must
39  /// provide this function.
40  ///
41  /// Example:
42  /// const RigAnimFb* MyLoadRigAnimFn(const char* anim_name,
43  /// std::string* scratch_buf) {
44  /// const bool load_ok = MyLoadFile(anim_name, scratch_buf);
45  /// assert(load_ok);
46  /// return motive::GetRigAnimFb(scratch_buf->c_str());
47  /// }
48  typedef const RigAnimFb* LoadRigAnimFn(const char* anim_name,
49  std::string* scratch_buf);
50 
51  /// Load the AnimTable specified in the FlatBuffer `params`.
52  /// For each animation in the AnimTable, `load_fn` is called to get the
53  /// `RigAnimFb` associated with the animation.
54  bool InitFromFlatBuffers(const AnimTableFb& params, LoadRigAnimFn* load_fn);
55 
56  /// Get an animation by index. This is fast and is the preferred way to
57  /// look up an animation.
58  /// @param object An enum defined by the caller specifying the object type.
59  /// For example, if you want separate animations lists for
60  /// cats and dogs, then object=0 could be cats, and object=1
61  /// could be dogs. The enum should run consecutively from
62  /// 0 to number of object types -1.
63  /// @param anim_idx An enum defined by the caller specifying the index into
64  /// the animation list for `object`. If your cat can run
65  /// and sleep, then you might make run=0 and sleep=1. Enum
66  /// should run consecutively from 0 to the number of
67  /// animations - 1. No need for the number of animations to
68  /// match between objects.
69  const RigAnim* Query(int object, int anim_idx) const {
70  const AnimIndex idx = CalculateIndex(object, anim_idx);
71  return idx == kInvalidAnimIndex ? nullptr : &anims_[idx];
72  }
73 
74  /// Get an animation by name. This is slow and should be avoided when
75  /// possible.
76  const RigAnim* QueryByName(const char* anim_name) const {
77  return const_cast<AnimTable*>(this)->QueryByName(anim_name);
78  }
79 
80  /// Return animation that defines the complete rig of this object.
81  const RigAnim& DefiningAnim(int object) const {
82  return defining_anims_[object];
83  }
84 
85  /// Return size of the top-most vector. Recall that AnimTable is a vector
86  /// of vectors.
87  int NumObjects() const { return static_cast<int>(indices_.size()); }
88 
89  private:
90  typedef uint16_t AnimIndex;
91  typedef std::vector<AnimIndex> AnimList;
92  typedef std::pair<std::string, AnimIndex> NameToIndex;
93  static const AnimIndex kInvalidAnimIndex = static_cast<AnimIndex>(-1);
94 
95  AnimIndex AddAnimName(const char* anim_name);
96  void InitNameMapFromFlatBuffers(const AnimTableFb& params);
97  bool LoadAnimations(LoadRigAnimFn* load_fn);
98  void AnimNames(std::vector<const char*>* anim_names) const;
99  size_t MaxAnimIndex() const;
100  size_t GatherObjectAnims(int object, const RigAnim** anims) const;
101  void CalculateDefinineAnims();
102 
103  RigAnim* QueryByName(const char* anim_name) {
104  auto map_entry = name_map_.find(anim_name);
105  if (map_entry == name_map_.end()) return nullptr;
106  return &anims_[map_entry->second];
107  }
108 
109  AnimIndex CalculateIndex(int object, int anim_idx) const {
110  assert(0 <= object && object < static_cast<int>(indices_.size()));
111  const AnimList& list = indices_[object];
112 
113  assert(0 <= anim_idx && anim_idx < static_cast<int>(list.size()));
114  return list[anim_idx];
115  }
116 
117  /// Vector of vectors. Top vector is indexed by `object`. Bottom vector is
118  /// indexed by `anim_idx`. Holds an index into the `anims_` array.
119  std::vector<AnimList> indices_;
120 
121  /// For each `object` type, holds the index to the animation that defines
122  /// the complete rig. Animations may animate only a sub-rig, but we need to
123  /// have a complete rig for initialization.
124  std::vector<RigAnim> defining_anims_;
125 
126  /// Map animation name to an animation index.
127  /// Used for by-name lookups, and to avoid duplicate copies of the same
128  /// animation.
129  std::unordered_map<std::string, AnimIndex> name_map_;
130 
131  /// Animation data. Contains no duplicate entries, thanks to name_map_.
132  std::vector<RigAnim> anims_;
133 };
