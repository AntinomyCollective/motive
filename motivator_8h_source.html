24 /// @class Motivator
25 /// @brief Drives a value towards a target value, or along a path.
26 ///
27 /// The value can be one-dimensional (e.g. a float), or multi-dimensional
28 /// (e.g. a matrix). The dimension is determined by the sub-class:
29 /// Motivator1f drives a float, MotivatorMatrix4f drives a 4x4 float matrix.
30 ///
31 /// Although you can instandiate a Motivator, you probably will not, since
32 /// there is no mechanism to read data out of a Motivator. Generally, you
33 /// will instantiate a derived class like Motivator1f, which provides
34 /// accessor functions.
35 ///
36 /// The way a Motivator's value moves towards its target is determined by the
37 /// **type** of a motivator. The type is specified in Motivator::Initialize().
38 ///
39 /// Note that a Motivator does not store any data itself. It is a handle into
40 /// a MotiveProcessor. Each MotiveProcessor holds all data for motivators
41 /// of its **type**.
42 ///
43 /// Only one Motivator can reference a specific index in a MotiveProcessor.
44 /// Therefore, when you copy a Motivator, the original motivator will become
45 /// invalid.
46 ///
47 class Motivator {
48  public:
49  Motivator() : processor_(nullptr), index_(kMotiveIndexInvalid) {}
50 
51  /// Allow Motivators to be copied.
52  /// Transfer ownership of motivator to new motivator. Old motivator is reset
53  /// and must be initialized again before being read.
54  /// We want to allow copies primarily so that we can have vectors of
55  /// Motivators.
56  Motivator(const Motivator& original) {
57  if (original.Valid()) {
58  original.processor_->TransferMotivator(original.index_, this);
59  } else {
60  processor_ = nullptr;
61  index_ = kMotiveIndexInvalid;
62  }
63  }
64 
65  /// Allow Motivators to be copied. `original` is reset.
66  /// See the copy constructor for details.
67  Motivator& operator=(const Motivator& original) {
68  Invalidate();
69  if (original.processor_ != nullptr) {
70  original.processor_->TransferMotivator(original.index_, this);
71  }
72  return *this;
73  }
74 
75  /// Remove ourselves from the MotiveProcessor when we're deleted.
76  ~Motivator() { Invalidate(); }
77 
78  /// Detatch this Motivator from its MotiveProcessor. Functions other than
79  /// Initialize() and Valid() can no longer be called afterwards.
80  void Invalidate() {
81  if (processor_ != nullptr) {
82  processor_->RemoveMotivator(index_);
83  }
84  }
85 
86  /// Return true if this Motivator is currently being driven by a
87  /// MotiveProcessor. That is, if it has been successfully initialized.
88  /// Also check for a consistent internal state.
89  bool Valid() const {
90  return processor_ != nullptr && processor_->ValidMotivator(index_, this);
91  }
92 
93  /// Return the type of Motivator we've been initilized to.
94  /// A Motivator can take on any type that matches its dimension.
95  /// The Motivator's type is determined by the `init` param in Initialize().
96  MotivatorType Type() const { return processor_->Type(); }
97 
98  /// The number of basic values that this Motivator is driving. For example,
99  /// a 3D position would return 3, since it drives three floats. A single
100  /// 4x4 matrix would return 1, since it's driving one matrix. The basic
101  /// value is determined by the MotiveProcessor backing this motivator.
102  MotiveDimension Dimensions() const { return processor_->Dimensions(index_); }
103 
104  protected:
105  Motivator(const MotivatorInit& init, MotiveEngine* engine,
106  MotiveDimension dimensions)
107  : processor_(nullptr), index_(kMotiveIndexInvalid) {
108  InitializeWithDimension(init, engine, dimensions);
109  }
110 
111  /// Initialize this Motivator to the type specified in init.type.
112  /// @param init Defines the type and initial state of the Motivator.
113  /// @param engine The engine that will update this Motivator when
114  /// engine->AdvanceFrame() is called.
115  /// @param dimensions The number of slots to occupy. For example, a 3D
116  /// position would occupy three slots.
117  void InitializeWithDimension(const MotivatorInit& init, MotiveEngine* engine,
118  MotiveDimension dimensions);
119 
120  /// The MotiveProcessor uses the functions below. It does not modify data
121  /// directly.
122  friend MotiveProcessor;
123 
124  /// These should only be called by MotiveProcessor!
125  void Init(MotiveProcessor* processor, MotiveIndex index) {
126  processor_ = processor;
127  index_ = index;
128  }
129  void Reset() { Init(nullptr, kMotiveIndexInvalid); }
130  const MotiveProcessor* Processor() const { return processor_; }
131 
132  /// All calls to an Motivator are proxied to an MotivatorProcessor. Motivator
133  /// data and processing is centralized to allow for scalable optimizations
134  /// (e.g. SIMD or parallelization).
135  MotiveProcessor* processor_;
136 
137  /// A MotiveProcessor processes one MotivatorType, and hosts every Motivator
138  /// of that type. This index here uniquely identifies this Motivator to the
139  /// MotiveProcessor.
140  MotiveIndex index_;
141 };

143 /// @class Motivator1f
144 /// @brief Drive a single float value towards a target, or along a spline.
145 ///
146 /// The current and target values and velocities can be specified by SetTarget()
147 /// or SetSpline().
148 ///
149 template <class VectorConverter, int kDimensionsT>
150 class MotivatorVectorTemplate : public Motivator {
151  public:
152  static const int kDimensions = kDimensionsT;
153  typedef VectorConverter C;
154  typedef typename fpl::ExternalVectorT<C, kDimensions>::type ExT;
155  typedef typename fpl::InternalVectorT<kDimensions>::type InT;
156  typedef typename MotiveTargetT<kDimensions>::type Target;
157  typedef fpl::SplinePlaybackN<kDimensions> Spline;
158  typedef MotiveTargetBuilderTemplate<C, kDimensions> TargetBuilder;
159 
160  /// Motivator is created in a reset state. When in the reset state,
161  /// it is not being driven, and Value(), Velocity(), etc. cannot be called.
162  MotivatorVectorTemplate() {}
163 
164  /// Initialize to the type specified by `init`. Current and target values
165  /// are not set.
166  MotivatorVectorTemplate(const MotivatorInit& init, MotiveEngine* engine)
167  : Motivator(init, engine, kDimensions) {}
168 
169  /// Initialize to the type specified by `init`. Set current and target values
170  /// as specified by `t`.
171  MotivatorVectorTemplate(const MotivatorInit& init, MotiveEngine* engine,
172  const Target& t)
173  : Motivator(init, engine, kDimensions) {
174  SetTarget(t);
175  }
176 
177  /// Initialize to the type specified by `init`.
178  void Initialize(const MotivatorInit& init, MotiveEngine* engine) {
179  InitializeWithDimension(init, engine, kDimensions);
180  }
181 
182  /// Initialize to the type specified by `init`. Set current and target values
183  /// as specified by `t`.
184  void InitializeWithTarget(const MotivatorInit& init, MotiveEngine* engine,
185  const Target& t) {
186  Initialize(init, engine);
187  SetTarget(t);
188  }
189 
190  /// Returns the current motivator value. The current value is updated when
191  /// engine->AdvanceFrame() is called on the `engine` that initialized this
192  /// Motivator.
193  /// Note that the "InT()" parameter is just a syntactic hack used to access
194  /// the correct overloaded function in the processor.
195  ExT Value() const { return C::To(Processor().ValueT(index_, InT())); }
196 
197  /// Returns the current rate of change of this motivator. For example,
198  /// if this Motivator is being driven by a spline, returns the derivative
199  /// at the current time in the spline curve.
200  ExT Velocity() const { return C::To(Processor().VelocityT(index_, InT())); }
201 
202  /// Returns the value this Motivator is driving towards.
203  /// If being driven by a spline, returns the value at the end of the spline.
204  ExT TargetValue() const {
205  return C::To(Processor().TargetValueT(index_, InT()));
206  }
207 
208  /// Returns the rate-of-change of this Motivator once it reaches
209  /// TargetValue().
210  ExT TargetVelocity() const {
211  return C::To(Processor().TargetVelocityT(index_, InT()));
212  }
213 
214  /// Returns TargetValue() minus Value(). If we're driving a
215  /// modular type (e.g. an angle), this may not be the naive subtraction.
216  /// For example, if TargetValue() = 170 degrees, Value() = -170 degrees,
217  /// then Difference() = -20 degrees.
218  ExT Difference() const {
219  return C::To(Processor().DifferenceT(index_, InT()));
220  }
221 
222  /// Returns time remaining until target is reached.
223  /// The unit of time is determined by the calling program.
224  MotiveTime TargetTime() const { return Processor().TargetTime(index_); }
225 
226  /// Returns the current time (i.e. the x-value) in the current spline.
227  /// If Motivator is not being driven by a spline, returns 0.
228  /// Whenever SetSpline() is called, this value will be reset to the
229  /// start_time specified in SplinePlayback. Every time
230  /// MotiveEngine::AdvanceFrame() is called, it will increment by
231  /// `delta_time` * `playback_rate`. If the SplinePlayback has repeat=true,
232  /// then SplineTime() will periodically loop back to time 0.
233  MotiveTime SplineTime() const { return Processor().SplineTime(index_); }
234 
235  /// Set the target and (optionally the current) motivator values.
236  /// Use this call to procedurally drive the Motivator towards a specific
237  /// target. The Motivator will transition smoothly to the new target.
238  /// You can change the target value every frame if you like, and the
239  /// Motivator value should behave calmly but responsively, with the
240  /// movement qualities of the underlying MotiveProcessor.
241  /// Note that the underlying MotiveProcessor is allowed to ignore
242  /// parts of `t` that are irrelevent to its algorithm.
243  /// @param t A set of waypoints to hit, optionally including the current
244  /// value. If the current value is not included, maintain the
245  /// existing current value.
246  void SetTarget(const Target& t) { Processor().SetTarget(index_, t); }
247 
248  /// Follow the curve specified in `s`. Overrides the existing current value.
249  /// @param s The spline to follow, the time in that spline to initiate
250  /// playback, and whether to repeat from the beginning after the
251  /// end of the spline is reached.
252  void SetSpline(const Spline& s) { Processor().SetSpline(index_, s); }
253 
254  /// Set rate at which we consume the spline set in SetSpline().
255  /// 0 ==> paused
256  /// 0.5 ==> half speed (slow motion)
257  /// 1 ==> authored speed
258  /// 2 ==> double speed (fast forward)
259  void SetSplinePlaybackRate(float playback_rate) {
260  Processor().SetSplinePlaybackRate(index_, playback_rate);
261  }
262 
263  private:
264  MotiveProcessorVector& Processor() {
265  return *static_cast<MotiveProcessorVector*>(processor_);
266  }
267  const MotiveProcessorVector& Processor() const {
268  return *static_cast<const MotiveProcessorVector*>(processor_);
269  }
270 };

272 /// @class MotivatorMatrix4fTemplate
273 /// @brief Drive a 4x4 float matrix from a series of basic transformations.
274 ///
275 /// The underlying basic transformations can be animated with
276 /// SetChildTarget1f(), and set to fixed values with SetChildValue1f() and
277 /// SetChildValue3f().
278 ///
279 /// Internally, we use mathfu::mat4 as our matrix type and mathfu::vec3 as
280 /// our vector type, but external we allow any matrix type to be specified
281 /// via the VectorConverter template parameter.
282 ///
283 template <class VectorConverter>
284 class MotivatorMatrix4fTemplate : public Motivator {
285  typedef VectorConverter C;
286  typedef typename VectorConverter::ExternalMatrix4 Mat4;
287  typedef typename VectorConverter::ExternalVector3 Vec3;
288 
289  public:
290  MotivatorMatrix4fTemplate() {}
291  MotivatorMatrix4fTemplate(const MotivatorInit& init, MotiveEngine* engine)
292  : Motivator(init, engine, 1) {}
293 
294  /// Initialize to the type specified by `init`.
295  void Initialize(const MotivatorInit& init, MotiveEngine* engine) {
296  InitializeWithDimension(init, engine, 1);
297  }
298 
299  /// Return the current value of the Motivator. The processor returns a
300  /// vector-aligned matrix, so the cast should be valid for any user-defined
301  /// matrix type.
302  const Mat4& Value() const { return C::To(Processor().Value(index_)); }
303 
304  /// Return the translation component of the matrix.
305  /// The matrix is a 3D affine transform, so the translation component is the
306  /// fourth column.
307  Vec3 Position() const {
308  return C::To(Processor().Value(index_).TranslationVector3D());
309  }
310 
311  /// Return the current value of the `child_index`th basic transform that
312  /// drives this matrix.
313  /// @param child_index The index into MatrixInit::ops(). The ops() array
314  /// is a series of basic transformation operations that
315  /// compose this matrix. Each basic transformation has
316  /// a current value. We gather this value here.
317  float ChildValue1f(MotiveChildIndex child_index) const {
318  return Processor().ChildValue1f(index_, child_index);
319  }
320 
321  /// Returns the current values of the basic transforms at indices
322  /// (child_index, child_index + 1, child_index + 2).
323  /// Useful when you drive all the (x,y,z) components of a translation, scale,
324  /// or rotation.
325  /// @param child_index The first index into MatrixInit::ops(). The value at
326  /// this index is returned in the x component of Vec3.
327  /// y gets child_index + 1's value, and
328  /// z gets child_index + 2's value.
329  Vec3 ChildValue3f(MotiveChildIndex child_index) const {
330  return C::To(Processor().ChildValue3f(index_, child_index));
331  }
332 
333  /// Set the target the 'child_index'th basic transform.
334  /// Each basic transform can be driven by a child motivator.
335  /// This call lets us control those child motivators.
336  /// @param child_index The index into the MatrixInit::ops() that was passed
337  /// into Initialize(). This operation must have been
338  /// initialized with a MotivatorInit, not a constant value.
339  /// @param t The target values for the basic transform to animate towards.
340  /// Also, optionally, the current value for it to jump to.
341  void SetChildTarget1f(MotiveChildIndex child_index, const MotiveTarget1f& t) {
342  Processor().SetChildTarget1f(index_, child_index, t);
343  }
344 
345  /// Set the constant value of a child. Each basic matrix transformation
346  /// can be driven by a constant value instead of a Motivator.
347  /// This call lets us set those constant values.
348  /// @param child_index The index into the MatrixInit::ops() that was passed
349  /// into Initialize(). This operation must have been
350  /// initialized with a constant value, not a MotivatorInit.
351  /// @param value The new constant value of this operation.
352  void SetChildValue1f(MotiveChildIndex child_index, float value) {
353  Processor().SetChildValue1f(index_, child_index, value);
354  }
355 
356  /// Set the constant values of the basic transforms at indices
357  /// (child_index, child_index + 1, child_index + 2).
358  /// @param child_index The first index into MatrixInit::ops(). The constant
359  /// value at this index is set to the x component of
360  /// `value`.
361  /// child_index + 1's constant is set to value.y, and
362  /// child_index + 2's constant is set to value.z.
363  void SetChildValue3f(MotiveChildIndex child_index, const Vec3& value) {
364  Processor().SetChildValue3f(index_, child_index, C::From(value));
365  }
366 
367  private:
368  MotiveProcessorMatrix4f& Processor() {
369  return *static_cast<MotiveProcessorMatrix4f*>(processor_);
370  }
371  const MotiveProcessorMatrix4f& Processor() const {
372  return *static_cast<const MotiveProcessorMatrix4f*>(processor_);
373  }
374 };
