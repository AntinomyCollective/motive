24 /// @class Motivator
25 /// @brief Drives a value towards a target value, or along a path.
26 ///
27 /// The value can be one-dimensional (e.g. a float), or multi-dimensional
28 /// (e.g. a matrix). The dimension is determined by the sub-class:
29 /// Motivator1f drives a float, MatrixMotivator4f drives a 4x4 float matrix.
30 ///
31 /// Although you can instantiate a Motivator, you probably will not, since
32 /// there is no mechanism to read data out of a Motivator. Generally, you
33 /// will instantiate a derived class like Motivator1f, which provides
34 /// accessor functions.
35 ///
36 /// The way a Motivator's value moves towards its target is determined by the
37 /// **type** of a motivator. The type is specified in Motivator::Initialize().
38 ///
39 /// Note that a Motivator does not store any data itself. It is a handle into
40 /// a MotiveProcessor. Each MotiveProcessor holds all data for motivators
41 /// of its **type**.
42 ///
43 /// Only one Motivator can reference a specific index in a MotiveProcessor.
44 /// Therefore, when you copy a Motivator, the original motivator will become
45 /// invalid.
46 ///
47 class Motivator {
48  public:
49  Motivator() : processor_(nullptr), index_(kMotiveIndexInvalid) {}
50 
51  /// Allow Motivators to be copied.
52  /// Transfer ownership of motivator to new motivator. Old motivator is reset
53  /// and must be initialized again before being read.
54  /// We want to allow copies primarily so that we can have vectors of
55  /// Motivators.
56  Motivator(const Motivator& original) {
57  if (original.Valid()) {
58  original.processor_->TransferMotivator(original.index_, this);
59  } else {
60  processor_ = nullptr;
61  index_ = kMotiveIndexInvalid;
62  }
63  }
64 
65  /// Allow Motivators to be copied. `original` is reset.
66  /// See the copy constructor for details.
67  Motivator& operator=(const Motivator& original) {
68  Invalidate();
69  if (original.processor_ != nullptr) {
70  original.processor_->TransferMotivator(original.index_, this);
71  }
72  return *this;
73  }
74 
75  /// Remove ourselves from the MotiveProcessor when we're deleted.
76  ~Motivator() { Invalidate(); }
77 
78  /// Detatch this Motivator from its MotiveProcessor. Functions other than
79  /// Initialize() and Valid() can no longer be called afterwards.
80  void Invalidate() {
81  if (processor_ != nullptr) {
82  processor_->RemoveMotivator(index_);
83  }
84  }
85 
86  /// Return true if this Motivator is currently being driven by a
87  /// MotiveProcessor. That is, if it has been successfully initialized.
88  /// Also check for a consistent internal state.
89  bool Valid() const {
90  return processor_ != nullptr && processor_->ValidMotivator(index_, this);
91  }
92 
93  /// Return the type of Motivator we've been initilized to.
94  /// A Motivator can take on any type that matches its dimension.
95  /// The Motivator's type is determined by the `init` param in Initialize().
96  MotivatorType Type() const { return processor_->Type(); }
97 
98  /// The number of basic values that this Motivator is driving. For example,
99  /// a 3D position would return 3, since it drives three floats. A single
100  /// 4x4 matrix would return 1, since it's driving one matrix. The basic
101  /// value is determined by the MotiveProcessor backing this motivator.
102  MotiveDimension Dimensions() const { return processor_->Dimensions(index_); }
103 
104  protected:
105  Motivator(const MotivatorInit& init, MotiveEngine* engine,
106  MotiveDimension dimensions)
107  : processor_(nullptr), index_(kMotiveIndexInvalid) {
108  InitializeWithDimension(init, engine, dimensions);
109  }
110 
111  /// Initialize this Motivator to the type specified in init.type.
112  /// @param init Defines the type and initial state of the Motivator.
113  /// @param engine The engine that will update this Motivator when
114  /// engine->AdvanceFrame() is called.
115  /// @param dimensions The number of slots to occupy. For example, a 3D
116  /// position would occupy three slots.
117  void InitializeWithDimension(const MotivatorInit& init, MotiveEngine* engine,
118  MotiveDimension dimensions);
119 
120  /// The MotiveProcessor uses the functions below. It does not modify data
121  /// directly.
122  friend MotiveProcessor;
123 
124  /// These should only be called by MotiveProcessor!
125  void Init(MotiveProcessor* processor, MotiveIndex index) {
126  processor_ = processor;
127  index_ = index;
128  }
129  void Reset() { Init(nullptr, kMotiveIndexInvalid); }
130  const MotiveProcessor* Processor() const { return processor_; }
131 
132  /// All calls to an Motivator are proxied to an MotivatorProcessor. Motivator
133  /// data and processing is centralized to allow for scalable optimizations
134  /// (e.g. SIMD or parallelization).
135  MotiveProcessor* processor_;
136 
137  /// A MotiveProcessor processes one MotivatorType, and hosts every Motivator
138  /// of that type. This index here uniquely identifies this Motivator to the
139  /// MotiveProcessor.
140  MotiveIndex index_;
141 };

143 /// @class MotivatorVectorTemplate
144 /// @brief Drive a tuple of floats towards a target, or along a spline.
145 ///
146 /// This template is instantiated below with four variations: 1~4 floats.
147 ///
148 /// The current and target values and velocities can be specified by SetTarget()
149 /// or SetSpline().
150 ///
151 template <class VectorConverter, MotiveDimension kDimensionsT>
152 class MotivatorVectorTemplate : public Motivator {
153  public:
154  static const MotiveDimension kDimensions = kDimensionsT;
155  typedef VectorConverter C;
156  typedef typename motive::ExternalVectorT<C, kDimensions>::type ExT;
157  typedef typename motive::InternalVectorT<kDimensions>::type InT;
158  typedef typename MotiveTargetT<kDimensions>::type Target;
159  typedef MotiveTargetBuilderTemplate<C, kDimensions> TargetBuilder;
160 
161  /// Motivator is created in a reset state. When in the reset state,
162  /// it is not being driven, and Value(), Velocity(), etc. cannot be called.
163  MotivatorVectorTemplate() {}
164 
165  /// Initialize to the type specified by `init`. Current and target values
166  /// are not set.
167  MotivatorVectorTemplate(const MotivatorInit& init, MotiveEngine* engine)
168  : Motivator(init, engine, kDimensions) {}
169 
170  /// Initialize to the type specified by `init`. Set current and target values
171  /// as specified by `t`.
172  MotivatorVectorTemplate(const MotivatorInit& init, MotiveEngine* engine,
173  const Target& t)
174  : Motivator(init, engine, kDimensions) {
175  SetTarget(t);
176  }
177 
178  /// Initialize to the type specified by `init`.
179  void Initialize(const MotivatorInit& init, MotiveEngine* engine) {
180  InitializeWithDimension(init, engine, kDimensions);
181  }
182 
183  /// Initialize to the type specified by `init`. Set current and target values
184  /// as specified by `t`.
185  void InitializeWithTarget(const MotivatorInit& init, MotiveEngine* engine,
186  const Target& t) {
187  Initialize(init, engine);
188  SetTarget(t);
189  }
190 
191  /// Returns the current motivator value. The current value is updated when
192  /// engine->AdvanceFrame() is called on the `engine` that initialized this
193  /// Motivator.
194  /// Note that the "InT()" parameter is just a syntactic hack used to access
195  /// the correct overloaded function in the processor.
196  ExT Value() const { return C::To(Processor().ValueT(index_, InT())); }
197 
198  /// Returns the current rate of change of this motivator. For example,
199  /// if this Motivator is being driven by a spline, returns the derivative
200  /// at the current time in the spline curve.
201  ExT Velocity() const { return C::To(Processor().VelocityT(index_, InT())); }
202 
203  /// Returns the velocity when playback rate is 1. Useful to know the
204  /// direction of a multi-dimensional motivator, even when playback rate
205  /// is 0.
206  ExT Direction() const { return C::To(Processor().DirectionT(index_, InT())); }
207 
208  /// Returns the value this Motivator is driving towards.
209  /// If being driven by a spline, returns the value at the end of the spline.
210  ExT TargetValue() const {
211  return C::To(Processor().TargetValueT(index_, InT()));
212  }
213 
214  /// Returns the rate-of-change of this Motivator once it reaches
215  /// TargetValue().
216  ExT TargetVelocity() const {
217  return C::To(Processor().TargetVelocityT(index_, InT()));
218  }
219 
220  /// Returns TargetValue() minus Value(). If we're driving a
221  /// modular type (e.g. an angle), this may not be the naive subtraction.
222  /// For example, if TargetValue() = 170 degrees, Value() = -170 degrees,
223  /// then Difference() = -20 degrees.
224  ExT Difference() const {
225  return C::To(Processor().DifferenceT(index_, InT()));
226  }
227 
228  /// Returns time remaining until target is reached.
229  /// The unit of time is determined by the calling program.
230  MotiveTime TargetTime() const { return Processor().TargetTime(index_); }
231 
232  /// Returns the current time (i.e. the x-value) in the current spline.
233  /// If Motivator is not being driven by a spline, returns 0.
234  /// Whenever SetSpline() is called, this value will be reset to the
235  /// start_time specified in SplinePlayback. Every time
236  /// MotiveEngine::AdvanceFrame() is called, it will increment by
237  /// `delta_time` * `playback_rate`. If the SplinePlayback has repeat=true,
238  /// then SplineTime() will periodically loop back to time 0.
239  MotiveTime SplineTime() const { return Processor().SplineTime(index_); }
240 
241  /// Set the target and (optionally the current) motivator values.
242  /// Use this call to procedurally drive the Motivator towards a specific
243  /// target. The Motivator will transition smoothly to the new target.
244  /// You can change the target value every frame if you like, and the
245  /// Motivator value should behave calmly but responsively, with the
246  /// movement qualities of the underlying MotiveProcessor.
247  /// Note that the underlying MotiveProcessor is allowed to ignore
248  /// parts of `t` that are irrelevent to its algorithm.
249  /// @param t A set of waypoints to hit, optionally including the current
250  /// value. If the current value is not included, maintain the
251  /// existing current value.
252  void SetTarget(const Target& t) { Processor().SetTarget(index_, t); }
253 
254  /// Follow the curve specified in `spline`. Overrides the existing current
255  /// value.
256  /// @param spline The spline to follow. Array of length Dimensions().
257  /// @param playback The time into the splines to initiate playback,
258  /// the blend time to the splines, and whether to repeat
259  /// from the beginning after the end of the spline is reached.
260  void SetSpline(const motive::CompactSpline& spline,
261  const motive::SplinePlayback& playback) {
262  assert(Dimensions() == 1);
263  Processor().SetSpline(index_, spline, playback);
264  }
265 
266  /// Follow the curves specified in `splines`. Overrides the existing current
267  /// value.
268  /// @param splines The splines that the curves should follow.
269  /// Array of length Dimensions().
270  /// @param playback The time into the splines to initiate playback,
271  /// the blend time to the splines, and whether to repeat
272  /// from the beginning after the end of the spline is reached.
273  void SetSplines(const motive::CompactSpline* splines,
274  const motive::SplinePlayback& playback) {
275  Processor().SetSplines(index_, splines, playback);
276  }
277 
278  void SetSplineTime(MotiveTime time) {
279  Processor().SetSplineTime(index_, time);
280  }
281 
282  /// Set rate at which we consume the spline set in SetSpline().
283  /// 0 ==> paused
284  /// 0.5 ==> half speed (slow motion)
285  /// 1 ==> authored speed
286  /// 2 ==> double speed (fast forward)
287  void SetSplinePlaybackRate(float playback_rate) {
288  Processor().SetSplinePlaybackRate(index_, playback_rate);
289  }
290 
291  private:
292  VectorProcessor& Processor() {
293  return *static_cast<VectorProcessor*>(processor_);
294  }
295  const VectorProcessor& Processor() const {
296  return *static_cast<const VectorProcessor*>(processor_);
297  }
298 };

300 /// @class MatrixMotivator4fTemplate
301 /// @brief Drive a 4x4 float matrix from a series of basic transformations.
302 ///
303 /// The underlying basic transformations can be animated with
304 /// SetChildTarget1f(), and set to fixed values with SetChildValue1f() and
305 /// SetChildValue3f().
306 ///
307 /// Internally, we use mathfu::mat4 as our matrix type and mathfu::vec3 as
308 /// our vector type, but external we allow any matrix type to be specified
309 /// via the VectorConverter template parameter.
310 ///
311 template <class VectorConverter>
312 class MatrixMotivator4fTemplate : public Motivator {
313  typedef VectorConverter C;
314  typedef typename VectorConverter::ExternalMatrix4 Mat4;
315  typedef typename VectorConverter::ExternalVector3 Vec3;
316  typedef MotivatorVectorTemplate<C, 1> Mot1f;
317 
318  public:
319  MatrixMotivator4fTemplate() {}
320  MatrixMotivator4fTemplate(const MotivatorInit& init, MotiveEngine* engine)
321  : Motivator(init, engine, 1) {}
322 
323  /// Initialize to the type specified by `init`.
324  void Initialize(const MotivatorInit& init, MotiveEngine* engine) {
325  InitializeWithDimension(init, engine, 1);
326  }
327 
328  /// Return the current value of the Motivator. The processor returns a
329  /// vector-aligned matrix, so the cast should be valid for any user-defined
330  /// matrix type.
331  const Mat4& Value() const { return C::To(Processor().Value(index_)); }
332 
333  /// Return the translation component of the matrix.
334  /// The matrix is a 3D affine transform, so the translation component is the
335  /// fourth column.
336  Vec3 Position() const {
337  return C::To(Processor().Value(index_).TranslationVector3D());
338  }
339 
340  /// Return the time remaining in the current spline animation.
341  /// Time units are defined by the user.
342  MotiveTime TimeRemaining() const { return Processor().TimeRemaining(index_); }
343 
344  /// Query the number of matrix operations. This equals the number of
345  /// operations in the `init` initializer.
346  int NumChildren() const { return Processor().NumChildren(index_); }
347 
348  /// Return the current value of the `child_index`th basic transform that
349  /// drives this matrix.
350  /// @param child_index The index into MatrixInit::ops(). The ops() array
351  /// is a series of basic transformation operations that
352  /// compose this matrix. Each basic transformation has
353  /// a current value. We gather this value here.
354  float ChildValue1f(MotiveChildIndex child_index) const {
355  return Processor().ChildValue1f(index_, child_index);
356  }
357 
358  /// Returns the current values of the basic transforms at indices
359  /// (child_index, child_index + 1, child_index + 2).
360  /// Useful when you drive all the (x,y,z) components of a translation, scale,
361  /// or rotation.
362  /// @param child_index The first index into MatrixInit::ops(). The value at
363  /// this index is returned in the x component of Vec3.
364  /// y gets child_index + 1's value, and
365  /// z gets child_index + 2's value.
366  Vec3 ChildValue3f(MotiveChildIndex child_index) const {
367  return C::To(Processor().ChildValue3f(index_, child_index));
368  }
369 
370  /// Returns the Motivator1f that's driving this child, if it's driven by
371  /// a Motivator1f. Otherwise, returns nullptr.
372  const Mot1f* ChildMotivator1f(MotiveChildIndex child_index) const {
373  return static_cast<const Mot1f*>(
374  Processor().ChildMotivator1f(index_, child_index));
375  }
376 
377  /// Set the target the 'child_index'th basic transform.
378  /// Each basic transform can be driven by a child motivator.
379  /// This call lets us control those child motivators.
380  /// @param child_index The index into the MatrixInit::ops() that was passed
381  /// into Initialize(). This operation must have been
382  /// initialized with a MotivatorInit, not a constant value.
383  /// @param t The target values for the basic transform to animate towards.
384  /// Also, optionally, the current value for it to jump to.
385  void SetChildTarget1f(MotiveChildIndex child_index, const MotiveTarget1f& t) {
386  Processor().SetChildTarget1f(index_, child_index, t);
387  }
388 
389  /// Set the constant value of a child. Each basic matrix transformation
390  /// can be driven by a constant value instead of a Motivator.
391  /// This call lets us set those constant values.
392  /// @param child_index The index into the MatrixInit::ops() that was passed
393  /// into Initialize(). This operation must have been
394  /// initialized with a constant value, not a MotivatorInit.
395  /// @param value The new constant value of this operation.
396  void SetChildValue1f(MotiveChildIndex child_index, float value) {
397  Processor().SetChildValue1f(index_, child_index, value);
398  }
399 
400  /// Set the constant values of the basic transforms at indices
401  /// (child_index, child_index + 1, child_index + 2).
402  /// @param child_index The first index into MatrixInit::ops(). The constant
403  /// value at this index is set to the x component of
404  /// `value`.
405  /// child_index + 1's constant is set to value.y, and
406  /// child_index + 2's constant is set to value.z.
407  /// @param value The new constant value for this 3-dimensional operation.
408  void SetChildValue3f(MotiveChildIndex child_index, const Vec3& value) {
409  Processor().SetChildValue3f(index_, child_index, C::From(value));
410  }
411 
412  /// Match existing MatrixOps with those in `ops` and smoothly transition
413  /// to the new parameters in `ops`.
414  void BlendToOps(const MatrixOpArray& ops,
415  const motive::SplinePlayback& playback) {
416  Processor().BlendToOps(index_, ops, playback);
417  }
418 
419  void SetPlaybackRate(float playback_rate) {
420  Processor().SetPlaybackRate(index_, playback_rate);
421  }
422 
423  private:
424  MatrixProcessor4f& Processor() {
425  return *static_cast<MatrixProcessor4f*>(processor_);
426  }
427  const MatrixProcessor4f& Processor() const {
428  return *static_cast<const MatrixProcessor4f*>(processor_);
429  }
430 };
