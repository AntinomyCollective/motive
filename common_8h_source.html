27 /// @typedef MotivatorType
28 /// MotivatorType is used for run-time type information. It's implemented as a
29 /// pointer to a string. Each derivation of MotivatorInit supplies a `kType`
30 /// static member that's identifies the corresponding MotiveProcessor.
31 /// We put `kType` in MotivatorInit instead of MotiveProcessor because
32 /// only MotivatorInit is in the external API.
33 typedef const char** MotivatorType;
34 static const MotivatorType kMotivatorTypeInvalid = nullptr;

36 /// @typedef MotiveIndex
37 /// The MotiveIndex identifies an Motivator inside a MotiveProcessor. The
38 /// MotiveProcessor holds all Motivators of its type. Calls to Motivators are
39 /// proxied to the MotiveProcessor, along with their MotiveIndex.
40 typedef int32_t MotiveIndex;
41 static const MotiveIndex kMotiveIndexInvalid = static_cast<MotiveIndex>(-1);

43 /// @typedef MotiveDimension
44 /// Identify how many slots in the MotiveProcessor a Motivator occupies.

46 /// between MotiveIndex and MotiveDimension in fplutil::IndexAllocator, so they
47 /// should be the same type.

63 /// @typedef BoneIndex
64 /// Identify bone for skeletal animation. Each non-root bone has a parent
65 /// whose BoneIndex is less than its own. Each bone has a transformation matrix.
66 /// By traversing up the tree to a root bone, multiplying the transformation
67 /// matrices as you go, you can get the global transform for the bone.
68 /// We support up to 254 bones.
69 typedef uint8_t BoneIndex;

71 static const BoneIndex kInvalidBoneIdx = 255;

73 /// @class MotivatorInit
74 /// Base class for Motivator parameterization. Every motivator type has a
75 /// its own init class that derives from MotivatorInit. In the derivation,
76 /// the MOTIVE_INTERFACE() macro should be added to the public interface.
77 /// This macro will define `kName`, `kType`, and Register() calls for the
78 /// motivator type.
79 class MotivatorInit {
80  public:
81  /// The derived class's constructor should set 'type'.
82  explicit MotivatorInit(MotivatorType type) : type_(type) {}
83 
84  MotivatorType type() const { return type_; }
85  void set_type(MotivatorType type) { type_ = type; }
86 
87  private:
88  MotivatorType type_;
89 };

91 /// Add this to the public interface of your derivation of Init. It defines
92 /// a unique identifier for this type as kType. Your derivation's constructor
93 /// should construct base class with Init(kType).
94 #define MOTIVE_INTERFACE() \
95  static const char* kName; \
96  static const motive::MotivatorType kType; \
97  static void Register()

99 /// Add this to the source file with your processor code. It instantiates the
100 /// static variables and functions declared in MOTIVE_INTERFACE.
101 /// Example usage,
102 /// MOTIVE_INSTANCE(AwesomeInit, AwesomeMotiveProcessor);
103 #define MOTIVE_INSTANCE(InitType, ProcessorType) \
104  static motive::MotiveProcessor* ProcessorType##Create() { \
105  return new ProcessorType(); \
106  } \
107  static void ProcessorType##Destroy(motive::MotiveProcessor* p) { delete p; } \
108  void InitType::Register() { \
109  const motive::MotiveProcessorFunctions functions(ProcessorType##Create, \
110  ProcessorType##Destroy); \
111  motive::MotiveEngine::RegisterProcessorFactory(InitType::kType, \
112  functions); \
113  } \
114  const char* InitType::kName = #ProcessorType; \
115  const motive::MotivatorType InitType::kType = &InitType::kName
