22 /// @class QuadraticEaseInEaseOut
23 /// @brief Represent a curve containing an ease in and ease out
24 /// quadratic curve.
25 ///
26 /// This class parameterizes two curves: ease-in followed by ease-out.
27 /// The ease-in curve ends after intersection_x, and the ease-out curve
28 /// matches the ease-in curve's value and derivative at that x. The
29 /// ease-in curve begins at x = 0. The ease-out curve ends at x = total_x.
30 class QuadraticEaseInEaseOut {
31  public:
32  QuadraticEaseInEaseOut() : intersection_x_(0.0f), total_x_(0.0f) {}
33  QuadraticEaseInEaseOut(const QuadraticCurve& in, const QuadraticCurve& out,
34  float intersection_x, float total_x)
35  : in_curve_(in),
36  out_curve_(out),
37  intersection_x_(intersection_x),
38  total_x_(total_x) {}
39  QuadraticEaseInEaseOut(const QuadraticCurve& in, float end_x)
40  : in_curve_(in),
41  out_curve_(in),
42  intersection_x_(end_x),
43  total_x_(end_x) {}
44 
45  /// Return the curve's value at x.
46  float Evaluate(float x) const { return Curve(x).Evaluate(x); }
47 
48  /// Return the curve's derivative at 'x'.
49  float Derivative(float x) const { return Curve(x).Derivative(x); }
50 
51  /// Return the curve's second derivative at 'x'
52  /// using either the in_curve or out_curve.
53  float SecondDerivative(float x) const { return Curve(x).SecondDerivative(x); }
54 
55  /// Return the quadratic function's third derivative: 0.
56  /// Even though `x` is unused, we pass it in for consistency with other
57  /// curve classes.
58  float ThirdDerivative(float x) const {
59  (void)x;
60  return 0.0f;
61  }
62 
63  /// Return the x at which in_curve and out_curve intersect.
64  float intersection_x() const { return intersection_x_; }
65 
66  /// Return the x where the curve reaches its end_value.
67  float total_x() const { return total_x_; }
68 
69  /// Return the in_curve.
70  const QuadraticCurve& in_curve() const { return in_curve_; }
71 
72  /// Return the out_curve.
73  const QuadraticCurve& out_curve() const { return out_curve_; }
74 
75  /// Return either the in_curve or the out_curve, depending
76  /// on if x is before or after intersection_x.
77  const QuadraticCurve& Curve(float x) const {
78  return x <= intersection_x_ ? in_curve_ : out_curve_;
79  }
80 
81  /// Equality. Checks for exact match. Useful for testing.
82  bool operator==(const QuadraticEaseInEaseOut& rhs) const {
83  return intersection_x_ == rhs.intersection_x_ && total_x_ == rhs.total_x_ &&
84  out_curve_ == rhs.out_curve_ && in_curve_ == rhs.in_curve_;
85  }
86  bool operator!=(const QuadraticEaseInEaseOut& rhs) const {
87  return !operator==(rhs);
88  }
89 
90  private:
91  QuadraticCurve in_curve_;
92  QuadraticCurve out_curve_;
93  float intersection_x_;
94  float total_x_;
95 };

289 /// @class QuadraticSpring
290 /// @brief An oscillating curve that accelerates quadratically.
291 ///
292 /// The curve can either,
293 /// - dampen (as in graph) if bias < 1,
294 /// - grow if bias > 1,
295 /// - or have constant amplitude if bias = 1.
296 ///
297 /// value
298 /// ^
299 /// |
300 /// typical delta value +--___ typical
301 /// | --_ delta x
302 /// | \_ .
303 /// | \ .
304 /// | \ .
305 /// | \. _--_
306 /// target value +-------------+---------+----+__+-----> x
307 /// | \_ _/
308 /// | -___-
309 ///
310 /// When starting from zero velocity, the curve will travel typical_delta_value
311 /// in typical_delta_x to intercept the target_value for the first time.
312 ///
313 /// However, the spring motion can start from any value and velocity, and the
314 /// calculated movement will mimic the typical momement mathematically.
315 /// The start state is specified in current_value and current_derivative.
316 ///
317 class QuadraticSpring {
318  public:
319  /// @brief Describe one portion of the QuadraticSpring curve. Useful for
320  /// quickly evaluating the curve iteratively.
321  /// @note This struct should not be used directly. Instead, it should be
322  /// passed into @ref EvaluateWithContext, @ref DerivativeWithContext,
323  /// and @ref SecondDerivativeWithContext.
324  struct Context {
325  Context()
326  : coeff(0.0f), peak(0.0f), peak_x(0.0f), valid_x(0.0f) {}
327 
328  // This constructor should only be called internally by
329  // @ref CalculateContext and @ref IncrementContext.
330  Context(float coeff, float peak, float peak_x, float valid_start_x,
331  float valid_end_x)
332  : coeff(coeff), peak(peak), peak_x(peak_x),
333  valid_x(valid_start_x, valid_end_x) {}
334 
335  // This diagram illustrates the meaning of each member variable.
336  // Note that this structure is only used internally, and should not be
337  // accessed or modified by external code. Explanations given for reference
338  // only.
339  //
340  // value C(x) = coeff * x^2 + peak
341  // ^ .
342  // | .
343  // peak +--___ .
344  // | --_ peak_x
345  // | \_ .
346  // | \ .
347  // | \ .
348  // | \.
349  // 0 +-------------+---> x
350  //
351  // <== valid_x ==>
352 
353  // The quadratic coefficient.
354  // If negative, curves downwards; if positive, curves upwards.
355  // Its absolute value will be either start_away_coeff_,
356  // if moving away from the target value, or start_away_coeff_ * bias_,
357  // if moving towards.
358  float coeff;
359 
360  // The amplitude of the current oscilation.
361  float peak;
362 
363  // The x value of the current oscillation.
364  // Either valid_x.start(), if curving towards the target, or valid_x.end(),
365  // if curving away.
366  // In "internal-x" values, where the start_peak_ is at x = 0.
367  float peak_x;
368 
369  // The x values that are valid for this segment of the spring curve.
370  // In "internal-x" values, where the start_peak_ is at x = 0.
371  Range valid_x;
372  };
373

398  /// @brief Creates a curve that starts at `current_value` and oscillates
399  /// about `target_value`.
400  /// @param current_value The starting value of the curve.
401  /// @param current_derivative The starting derivative of the curve.
402  /// @param target_value The value we oscillate about.
403  /// @param typical_delta_value Together with typical_delta_x, describes how
404  /// quickly the curve moves to the target.
405  /// See class description for details.
406  /// @param typical_delta_x Together with typical_delta_value, describes how
407  /// quickly the curve moves to the target.
408  /// See class description for details.
409  /// @param bias Determines how quickly the curve settles down. That is, the
410  /// tightness of the spring. The smaller, the tighter.
411  /// < 1 ==> dampens down to zero eventually
412  /// = 1 ==> oscilates with same amplitude indefinitely
413  /// > 1 ==> amplitude grows with every oscillation
414  /// Each peak has has the magnitude of the previous peak * bias.
415  QuadraticSpring(float current_value, float current_derivative,
416  float target_value, float typical_delta_value,
417  float typical_delta_x, float bias);

419  /// @brief Ensure Context `c` is valid for `external_x`.
420  ///
421  /// Most times, `external_x` will already be within the valid x-range of `c`.
422  /// When it's not, we advance `c` to the next quadratic in the series of
423  /// curves that compose the spring curve. We continue until `external_x` is
424  /// again in the valid range of `c`.
425  ///
426  /// @param external_x The current x value we're evaluating. The expectation is
427  /// that `external_x` will be incrementing at a reasonably
428  /// small and steady rate. When this happens, `c` only has
429  /// to be updated when `external_x` exceeds the portion of
430  /// the spring curve that `c` describes.
431  /// @param c The current evaluation context. It describes one portion of
432  /// the spring curve. After updating `c` with this function, you can
433  /// pass it into @ref EvaluateWithContext and other such functions
434  /// to calculate the curve values quickly.
435  void IncrementContext(float external_x, Context* c) const {
436  const float x = ToInternalX(external_x);
437  if (x > c->valid_x.end()) {
438  IncrementContextInternal(x, c);
439  }
440  assert(ValidX(external_x, *c));
441  }
442 
443  /// @brief Calculate a Context that describes the portion of the spring curve
444  /// near `external_x`.
445  ///
446  /// It's significantly faster to call this function only once, and then,
447  /// for succeeding xs, advance the Context by calling @ref IncrementContext.
448  ///
449  /// @param external_x The x value that about which we want the returned
450  /// Context to be valid.
451  /// @returns A Context that describes a portion of the spring curve.
452  /// Use it in calls to @ref EvaluateWithContext, etc.
453  Context CalculateContext(float external_x) const;
454 
455  /// @brief Calculate the spring curve value at `external_x`.
456  ///
457  /// @note This function is significantly slower than @ref EvaluateWithContext.
458  /// Consider maintaining a Context externally and then calling
459  /// @ref IncrementContext and @ref EvaluateWithContext instead of
460  /// just Evaluate.
461  ///
462  /// @param external_x x value since the start of the curve, as specified by
463  /// `current_value` and `current_derivative` in the
464  /// constructor.
465  /// @returns The spring curve value at `external_x`.
466  float Evaluate(float external_x) const {
467  const Context c = CalculateContext(external_x);
468  return EvaluateWithContext(external_x, c);
469  }
470 
471  /// @brief Calculate the spring curve derivative at `external_x`.
472  ///
473  /// @note This function is significantly slower than
474  /// @ref DerivativeWithContext.
475  /// Consider maintaining a Context externally and then calling
476  /// @ref IncrementContext and @ref DerivativeWithContext instead of
477  /// just Derivative.
478  ///
479  /// @param external_x x value since the start of the curve.
480  /// @returns The spring curve derivative at `external_x`.
481  float Derivative(float external_x) const {
482  const Context c = CalculateContext(external_x);
483  return DerivativeWithContext(external_x, c);
484  }
485 
486  /// @brief Calculate the spring curve second derivative at `external_x`.
487  ///
488  /// @note This function is significantly slower than
489  /// @ref SecondDerivativeWithContext.
490  /// Consider maintaining a Context externally and then calling
491  /// @ref IncrementContext and @ref SecondDerivativeWithContext
492  /// instead of just SecondDerivative.
493  ///
494  /// @param external_x x value since the start of the curve.
495  /// @returns The spring curve second derivative at `external_x`.
496  /// Note that there are only two possible second derivatives:
497  /// one towards the target and one away from the target.
498  /// This is a concequence of our usage of quadratic functions.
499  float SecondDerivative(float external_x) const {
500  const Context c = CalculateContext(external_x);
501  return SecondDerivativeWithContext(external_x, c);
502  }
503 
504  /// @brief Quickly calculate the spring curve value at `external_x`.
505  ///
506  /// @param external_x x value since the start of the curve.
507  /// @param c Context that describes the portion of the spring curve around
508  /// `external_x`. Note that you should call @ref IncrementContext
509  /// with this `external_x` and `c` before calling this function.
510  /// @returns The spring curve value at `external_x`.
511  float EvaluateWithContext(float external_x, const Context& c) const {
512  assert(ValidX(external_x, c));
513  const float quadratic_x = QuadraticX(external_x, c);
514  const float offset = c.coeff * quadratic_x * quadratic_x;
515  return target_ + c.peak + offset;
516  }
517 
518  /// @brief Quickly calculate the spring curve derivative at `external_x`.
519  ///
520  /// @param external_x x value since the start of the curve.
521  /// @param c Context that describes the portion of the spring curve around
522  /// `external_x`. Note that you should call @ref IncrementContext
523  /// with this `external_x` and `c` before calling this function.
524  /// @returns The spring curve derivative at `external_x`.
525  float DerivativeWithContext(float external_x, const Context& c) const {
526  assert(ValidX(external_x, c));
527  const float quadratic_x = QuadraticX(external_x, c);
528  return 2.0f * c.coeff * quadratic_x;
529  }
530 
531  /// @brief Quickly calculate the spring curve second derivative at
532  /// `external_x`.
533  ///
534  /// @param external_x x value since the start of the curve.
535  /// @param c Context that describes the portion of the spring curve around
536  /// `external_x`. Note that you should call @ref IncrementContext
537  /// with this `external_x` and `c` before calling this function.
538  /// @returns The spring curve second derivative at `external_x`.
539  float SecondDerivativeWithContext(float external_x, const Context& c) const {
540  assert(ValidX(external_x, c));
541  (void)external_x;
542  return 2.0f * c.coeff;
543  }
544 
545  /// @brief Return the spring curve's third derivative.
546  /// @returns Return value is always 0 because internally we return quadratics.
547  float ThirdDerivative(float external_x) const {
548  (void)external_x;
549  return 0.0f;
550  }
551 
552  /// @brief Calculate the x for the `iterations`th peak of the spring curve.
553  /// @param iterations The number of iterations to be completed, where an
554  /// iteration is defined as one peak to the next.
555  /// @returns x for the `iterations`th peak.
556  float IterationX(float iterations) const;
557 
558  /// @brief Get the `target_value` originally passed into the constructor.
559  /// @returns The value about which we're oscillating.
560  float target() const { return target_; }
561 
562  private:
563  bool ValidX(float external_x, const Context& c) const {
564  const float kErrorToleranceForX = 0.01f;
565  return c.valid_x.ContainsWithTolerance(ToInternalX(external_x),
566  kErrorToleranceForX);
567  }
568  void IncrementContextInternal(float x, Context* c) const;
569  float ToInternalX(float external_x) const { return external_x + start_x_; }
570  float ToExternalX(float internal_x) const { return internal_x - start_x_; }
571  float QuadraticX(float external_x, const Context& c) const {
572  return ToInternalX(external_x) - c.peak_x;
573  }
574 
575  // Quadratic coefficient when moving away from the target value (i.e. when
576  // spring is slowing down). Note that coefficient when moving towards target
577  // value is start_away_coeff_ * bias_.
578  float start_away_coeff_;
579 
580  // Delta between peak and target of the first peak in the curve.
581  float start_peak_;
582 
583  // Time of the start_peak_, above. Note that this could be a negative x,
584  // if the 'current' state in the constructor has the curve already moving
585  // towards the target.
586  float start_x_;
587 
588  // Target value we're oscillating about.
589  float target_;
590 
591  // Amount that each succeeding peak is dampened by.
592  // That is, the second peak will have amplitude start_peak_ * bias_.
593  float bias_;
594 
595  // Cached variable holding,
596  // R = sqrt(D/a) * (1 + b) / (sqrt(b) - b), when bias != 1 (see equation 5),
597  // and
598  // R = sqrt(a / D) / 2, when bias = 1 (see equation 8 in curve_util.cpp).
599  float r_;
600 
601  // Cached variable holding sqrt(bias_). This is expensive to calculate and
602  // commonly used.
603  float sqrt_b_;
604 
605  // Cached variable holding 1.0f / log(sqrt_b_).
606  float log_sqrt_b_reciprocal_;
607 };
