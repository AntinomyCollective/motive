22 /// @class MotiveNode1f
23 /// @brief A waypoint in MotiveTarget1f.
24 /// Describes one key point through which a value is animated.
25 struct MotiveNode1f {
26  float value; /// Desired value to animate to at `time`.
27  float velocity; /// Speed when at `time`.
28  MotiveTime time; /// Time to achieve this key point.
29  fpl::ModularDirection direction; /// When using modular arithmetic, which of
30  /// two directions to go.
31  MotiveNode1f()
32  : value(0.0f),
33  velocity(0.0f),
34  time(0),
35  direction(fpl::kDirectionClosest) {}
36  MotiveNode1f(float value, float velocity, MotiveTime time,
37  fpl::ModularDirection direction = fpl::kDirectionClosest)
38  : value(value), velocity(velocity), time(time), direction(direction) {}
39 };

41 /// @class MotiveTarget1f
42 /// @brief Set the current and/or target state for a one-dimensional Motivator.
43 ///
44 /// A series of waypoints through which we animate. If the first waypoint has
45 /// time = 0, the current value and velocity jumps to that waypoint's value and
46 /// velocity.
47 ///
48 /// MotiveTarget1fs are most easily created with the utility functions below,
49 /// for example Current1f, Target1f, CurrentToTarget1f.
50 ///
51 /// If the current value and velocity are not specified (i.e. if the first
52 /// waypoint has time > 0), then the current value and velocity in
53 /// the Motivator are maintained.
54 ///
55 /// If the target is not specified (i.e. only one waypoint which has time = 0),
56 /// then the current value is set as specified, and the velocity is set to 0.
57 ///
58 class MotiveTarget1f {
59  public:
60  static const int kMaxNodes = 3;
61 
62  MotiveTarget1f() : num_nodes_(0) {}
63 
64  /// Create with only one waypoint.
65  /// If n0.time = 0, set the current value and velocity.
66  /// If n0.time > 0, maintain the current value and velocity and animate to
67  /// n0's value and velocity in n0.time.
68  explicit MotiveTarget1f(const MotiveNode1f& n0) : num_nodes_(1) {
69  nodes_[0] = n0;
70  }
71 
72  /// Create with two waypoints.
73  /// Can be current to target, if n0.time = 0.
74  /// Or can maintain the current and to through two targets: first n0, then n1.
75  /// Precondition: n0.time < n1.time.
76  MotiveTarget1f(const MotiveNode1f& n0, const MotiveNode1f& n1)
77  : num_nodes_(2) {
78  assert(0 <= n0.time && n0.time < n1.time);
79  nodes_[0] = n0;
80  nodes_[1] = n1;
81  }
82 
83  /// Create with three waypoints.
84  /// 0 <= n0.time < n1.time < n2.time
85  MotiveTarget1f(const MotiveNode1f& n0, const MotiveNode1f& n1,
86  const MotiveNode1f& n2)
87  : num_nodes_(3) {
88  assert(0 <= n0.time && n0.time < n1.time && n1.time < n2.time);
89  nodes_[0] = n0;
90  nodes_[1] = n1;
91  nodes_[2] = n2;
92  }
93 
94  /// Empty the target of all waypoints.
95  void Reset() { num_nodes_ = 0; }
96 
97  /// Return nth waypoint.
98  /// @param node_index nth waypoint. 0 <= node_index < num_nodes()
99  const MotiveNode1f& Node(int node_index) const {
100  assert(0 <= node_index && node_index < num_nodes_);
101  return nodes_[node_index];
102  }
103 
104  /// Return smallest range that covers the values of all waypoints.
105  /// @param start_value An extra value to include in the min/max calculation.
106  /// Most often is the current value of the Motivator1f.
107  fpl::Range ValueRange(float start_value) const {
108  assert(num_nodes_ > 0);
109  float min = start_value;
110  float max = start_value;
111  for (int i = 0; i < num_nodes_; ++i) {
112  min = std::min(nodes_[i].value, min);
113  max = std::max(nodes_[i].value, max);
114  }
115  return fpl::Range(min, max);
116  }
117 
118  /// Return time of the last waypoint.
119  MotiveTime EndTime() const {
120  assert(num_nodes_ > 0);
121  return nodes_[num_nodes_ - 1].time;
122  }
123 
124  int num_nodes() const { return num_nodes_; }
125 
126  private:
127  /// Length of nodes_.
128  int num_nodes_;
129 
130  /// Constant-size array, to avoid dynamic memory allocation.
131  /// This class is often used as a parameter and allocated on the stack.
132  MotiveNode1f nodes_[kMaxNodes];
133 };

135 /// Set the Motivator's current values. Target values are reset to be the same
136 /// as the new current values.
137 inline MotiveTarget1f Current1f(float current_value,
138  float current_velocity = 0.0f) {
139  return MotiveTarget1f(MotiveNode1f(current_value, current_velocity, 0));
140 }
141 
142 /// Keep the Motivator's current values, but set the Motivator's target values.
143 /// If Motivator uses modular arithmetic, traverse from the current to the
144 /// target according to 'direction'.
145 inline MotiveTarget1f Target1f(
146  float target_value, float target_velocity, MotiveTime target_time,
147  fpl::ModularDirection direction = fpl::kDirectionClosest) {
148  assert(target_time > 0);
149  return MotiveTarget1f(
150  MotiveNode1f(target_value, target_velocity, target_time, direction));
151 }
152 
153 /// Set both the current and target values for an Motivator.
154 inline MotiveTarget1f CurrentToTarget1f(
155  float current_value, float current_velocity, float target_value,
156  float target_velocity, MotiveTime target_time,
157  fpl::ModularDirection direction = fpl::kDirectionClosest) {
158  return MotiveTarget1f(
159  MotiveNode1f(current_value, current_velocity, 0),
160  MotiveNode1f(target_value, target_velocity, target_time, direction));
161 }
162 
163 /// Move from the current value to the target value at a constant speed.
164 inline MotiveTarget1f CurrentToTargetConstVelocity1f(float current_value,
165  float target_value,
166  MotiveTime target_time) {
167  assert(target_time > 0);
168  const float velocity = (target_value - current_value) / target_time;
169  return MotiveTarget1f(
170  MotiveNode1f(current_value, velocity, 0),
171  MotiveNode1f(target_value, velocity, target_time, fpl::kDirectionDirect));
172 }

174 /// Keep the Motivator's current values, but set two targets for the Motivator.
175 /// After the first target, go on to the next.
176 inline MotiveTarget1f TargetToTarget1f(float target_value,
177  float target_velocity,
178  MotiveTime target_time,
179  float third_value, float third_velocity,
180  MotiveTime third_time) {
181  return MotiveTarget1f(
182  MotiveNode1f(target_value, target_velocity, target_time),
183  MotiveNode1f(third_value, third_velocity, third_time));
184 }
185 
186 /// Set the Motivator's current values, and two targets afterwards.
187 inline MotiveTarget1f CurrentToTargetToTarget1f(
188  float current_value, float current_velocity, float target_value,
189  float target_velocity, MotiveTime target_time, float third_value,
190  float third_velocity, MotiveTime third_time) {
191  return MotiveTarget1f(
